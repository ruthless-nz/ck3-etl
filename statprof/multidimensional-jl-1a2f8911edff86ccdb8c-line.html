16 """
17 CartesianIndex(i, j, k...) -> I
18 CartesianIndex((i, j, k...)) -> I
19
20 Create a multidimensional index `I`, which can be used for
21 indexing a multidimensional array `A`. In particular, `A[I]` is
22 equivalent to `A[i,j,k...]`. One can freely mix integer and
23 `CartesianIndex` indices; for example, `A[Ipre, i, Ipost]` (where
24 `Ipre` and `Ipost` are `CartesianIndex` indices and `i` is an
25 `Int`) can be a useful expression when writing algorithms that
26 work along a single dimension of an array of arbitrary
27 dimensionality.
28
29 A `CartesianIndex` is sometimes produced by [`eachindex`](@ref), and
30 always when iterating with an explicit [`CartesianIndices`](@ref).
31
32 # Examples
33 ```jldoctest
34 julia> A = reshape(Vector(1:16), (2, 2, 2, 2))
35 2×2×2×2 Array{Int64,4}:
36 [:, :, 1, 1] =
37 1 3
38 2 4
39
40 [:, :, 2, 1] =
41 5 7
42 6 8
43
44 [:, :, 1, 2] =
45 9 11
46 10 12
47
48 [:, :, 2, 2] =
49 13 15
50 14 16
51
52 julia> A[CartesianIndex((1, 1, 1, 1))]
53 1
54
55 julia> A[CartesianIndex((1, 1, 1, 2))]
56 9
57
58 julia> A[CartesianIndex((1, 1, 2, 1))]
59 5
60 ```

1710 function sortslices(A::AbstractArray; dims::Union{Integer, Tuple{Vararg{Integer}}}, kws...)
1711 _sortslices(A, Val{dims}(); kws...)
1712 end
1713
1714 # Works around inference's lack of ability to recognize partial constness
1715 struct DimSelector{dims, T}
1716 A::T
1717 end
1718 DimSelector{dims}(x::T) where {dims, T} = DimSelector{dims, T}(x)
1719 (ds::DimSelector{dims, T})(i) where {dims, T} = i in dims ? axes(ds.A, i) : (:,)
1720
1721 _negdims(n, dims) = filter(i->!(i in dims), 1:n)
1722
1723 function compute_itspace(A, ::Val{dims}) where {dims}
1724 negdims = _negdims(ndims(A), dims)
1725 axs = Iterators.product(ntuple(DimSelector{dims}(A), ndims(A))...)
1726 vec(permutedims(collect(axs), (dims..., negdims...)))
1727 end
1728
1729 function _sortslices(A::AbstractArray, d::Val{dims}; kws...) where dims
1730 itspace = compute_itspace(A, d)
1731 vecs = map(its->view(A, its...), itspace)
1732 p = sortperm(vecs; kws...)
1733 if ndims(A) == 2 && isa(dims, Integer) && isa(A, Array)
1734 # At the moment, the performance of the generic version is subpar
1735 # (about 5x slower). Hardcode a fast-path until we're able to
1736 # optimize this.
1737 return dims == 1 ? A[p, :] : A[:, p]
1738 else
1739 B = similar(A)
1740 for (x, its) in zip(p, itspace)
1741 B[its...] = vecs[x]
1742 end
1743 B
1744 end
1745 end
