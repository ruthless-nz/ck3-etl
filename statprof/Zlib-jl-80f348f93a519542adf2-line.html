361 function eof(r::Reader)
362 # Detecting EOF is somewhat tricky: we might not have reached
363 # EOF in r.io but decompressing the remaining data might
364 # yield no uncompressed data. So, make sure we can get at least
365 # one more byte of decompressed data before we say we haven't
366 # reached EOF yet.
367 bytesavailable(r.buf) == 0 && eof(r.io)
