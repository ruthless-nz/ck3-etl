33 For example, if a group of workers are executing several tasks, and multiple workers fail, the resulting `CompositeException` will
34 contain a "bundle" of information from each worker indicating where and why the exception(s) occurred.

173 function task_local_storage(body::Function, key, val)
174 tls = task_local_storage()
175 hadkey = haskey(tls, key)
176 old = get(tls, key, nothing)
177 tls[key] = val
178 try
179 return body()
180 finally
181 hadkey ? (tls[key] = old) : delete!(tls, key)

428 """
429 schedule(t::Task, [val]; error=false)
430
431 Add a [`Task`](@ref) to the scheduler's queue. This causes the task to run constantly when the system
432 is otherwise idle, unless the task performs a blocking operation such as [`wait`](@ref).
433
434 If a second argument `val` is provided, it will be passed to the task (via the return value of
435 [`yieldto`](@ref)) when it runs again. If `error` is `true`, the value is raised as an exception in
436 the woken task.
437
438 # Examples
439 ```jldoctest
440 julia> a5() = sum(i for i in 1:1000);
441
442 julia> b = Task(a5);
443
444 julia> istaskstarted(b)
445 false
446
447 julia> schedule(b);

449 julia> yield();
450
451 julia> istaskstarted(b)
452 true
453
454 julia> istaskdone(b)
455 true
456 ```
457 """
458 function schedule(t::Task, @nospecialize(arg); error=false)
459 # schedule a task to be (re)started with the given value or exception
460 t.state == :runnable || Base.error("schedule: Task not runnable")
461 if error
462 t.queue === nothing || Base.list_deletefirst!(t.queue, t)
463 t.exception = arg
464 else
465 t.queue === nothing || Base.error("schedule: Task not runnable")
466 t.result = arg
467 end
468 enq_work(t)
469 return t
470 end
