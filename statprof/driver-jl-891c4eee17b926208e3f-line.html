23 function normalize_expr(stmt::Expr)
24 if stmt.head === :gotoifnot
25 return GotoIfNot(stmt.args[1], stmt.args[2]::Int)
26 elseif stmt.head === :return
27 return (length(stmt.args) == 0) ? ReturnNode(nothing) : ReturnNode(stmt.args[1])
28 elseif stmt.head === :unreachable
29 return ReturnNode()
30 else
31 return stmt
32 end
33 end
