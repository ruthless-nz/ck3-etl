272 """
273 mapcols(f::Union{Function,Type}, df::AbstractDataFrame)
274
275 Return a `DataFrame` where each column of `df` is transformed using function `f`.
276 `f` must return `AbstractVector` objects all with the same length or scalars
277 (all values other than `AbstractVector` are considered to be a scalar).
278
279 Note that `mapcols` guarantees not to reuse the columns from `df` in the returned
280 `DataFrame`. If `f` returns its argument then it gets copied before being stored.
281
282 # Examples
283 ```jldoctest
284 julia> df = DataFrame(x=1:4, y=11:14)
285 4×2 DataFrame
286 │ Row │ x │ y │
287 │ │ Int64 │ Int64 │
288 ├─────┼───────┼───────┤
289 │ 1 │ 1 │ 11 │
290 │ 2 │ 2 │ 12 │
291 │ 3 │ 3 │ 13 │
292 │ 4 │ 4 │ 14 │
293
294 julia> mapcols(x -> x.^2, df)
295 4×2 DataFrame
296 │ Row │ x │ y │
297 │ │ Int64 │ Int64 │
298 ├─────┼───────┼───────┤
299 │ 1 │ 1 │ 121 │
300 │ 2 │ 4 │ 144 │
301 │ 3 │ 9 │ 169 │
302 │ 4 │ 16 │ 196 │
303 ```
304 """
305 function mapcols(f::Union{Function,Type}, df::AbstractDataFrame)
306 # note: `f` must return a consistent length
307 vs = AbstractVector[]
308 seenscalar = false
309 seenvector = false
310 for v in eachcol(df)
311 fv = f(v)
312 if fv isa AbstractVector
313 if seenscalar
314 throw(ArgumentError("mixing scalars and vectors in mapcols not allowed"))
315 end
316 seenvector = true
317 push!(vs, fv === v ? copy(fv) : fv)
318 else
319 if seenvector
320 throw(ArgumentError("mixing scalars and vectors in mapcols not allowed"))
321 end
322 seenscalar = true
323 push!(vs, [fv])
324 end
325 end
326 return DataFrame(vs, _names(df), copycols=false)
327 end
