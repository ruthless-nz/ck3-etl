94 function mapfoldr_impl(f, op, nt::NamedTuple{(:init,)}, itr, i::Integer)
95 init = nt.init
96 # Unroll the while loop once; if init is known, the call to op may
97 # be evaluated at compile time
98 if isempty(itr) || i == 0
99 return init
100 else
101 x = itr[i]
102 v = op(f(x), init)
103 while i > 1
104 x = itr[i -= 1]
105 v = op(f(x), v)
106 end
107 return v
108 end

146 ## reduce & mapreduce

148 # `mapreduce_impl()` is called by `mapreduce()` (via `_mapreduce()`, when `A`
149 # supports linear indexing) and does actual calculations (for `A[ifirst:ilast]` subset).
150 # For efficiency, no parameter validity checks are done, it's the caller's responsibility.
151 # `ifirst:ilast` range is assumed to be a valid non-empty subset of `A` indices.
152
153 # This is a generic implementation of `mapreduce_impl()`,
154 # certain `op` (e.g. `min` and `max`) may have their own specialized versions.
155 @noinline function mapreduce_impl(f, op, A::AbstractArray, ifirst::Integer, ilast::Integer, blksize::Int)
156 if ifirst == ilast
157 @inbounds a1 = A[ifirst]
158 return mapreduce_first(f, op, a1)
159 elseif ifirst + blksize > ilast
160 # sequential portion
161 @inbounds a1 = A[ifirst]
162 @inbounds a2 = A[ifirst+1]
163 v = op(f(a1), f(a2))
164 @simd for i = ifirst + 2 : ilast
165 @inbounds ai = A[i]
166 v = op(v, f(ai))
167 end
168 return v
169 else
170 # pairwise portion
171 imid = (ifirst + ilast) >> 1
172 v1 = mapreduce_impl(f, op, A, ifirst, imid, blksize)
173 v2 = mapreduce_impl(f, op, A, imid+1, ilast, blksize)
174 return op(v1, v2)
175 end
176 end

181 """
182 mapreduce(f, op, itrs...; [init])
183
184 Apply function `f` to each element(s) in `itrs`, and then reduce the result using the binary
185 function `op`. If provided, `init` must be a neutral element for `op` that will be returned
186 for empty collections. It is unspecified whether `init` is used for non-empty collections.
187 In general, it will be necessary to provide `init` to work with empty collections.
188
189 [`mapreduce`](@ref) is functionally equivalent to calling
190 `reduce(op, map(f, itr); init=init)`, but will in general execute faster since no
191 intermediate collection needs to be created. See documentation for [`reduce`](@ref) and
192 [`map`](@ref).
193
194 !!! compat "Julia 1.2"
195 `mapreduce` with multiple iterators requires Julia 1.2 or later.
196
197 # Examples
198 ```jldoctest
199 julia> mapreduce(x->x^2, +, [1:3;]) # == 1 + 4 + 9
200 14
201 ```
202
203 The associativity of the reduction is implementation-dependent. Additionally, some
204 implementations may reuse the return value of `f` for elements that appear multiple times in
205 `itr`. Use [`mapfoldl`](@ref) or [`mapfoldr`](@ref) instead for
206 guaranteed left or right associativity and invocation of `f` for every value.

305 function _mapreduce(f, op, ::IndexLinear, A::AbstractArray{T}) where T
306 inds = LinearIndices(A)
307 n = length(inds)
308 if n == 0
309 return mapreduce_empty(f, op, T)
310 elseif n == 1
311 @inbounds a1 = A[first(inds)]
312 return mapreduce_first(f, op, a1)
313 elseif n < 16 # process short array here, avoid mapreduce_impl() compilation
314 @inbounds i = first(inds)
315 @inbounds a1 = A[i]
316 @inbounds a2 = A[i+=1]
317 s = op(f(a1), f(a2))
318 while i < last(inds)
319 @inbounds Ai = A[i+=1]
320 s = op(s, f(Ai))
321 end
322 return s
323 else
324 return mapreduce_impl(f, op, A, first(inds), last(inds))
325 end
326 end

336 initial value `init` must be a neutral element for `op` that will be returned for empty

338

344 should be used instead: `maximum(itr)`, `minimum(itr)`, `sum(itr)`, `prod(itr)`,
345 `any(itr)`, `all(itr)`.
346

348 use non-associative operations like `-` because it is undefined whether `reduce(-,[1,2,3])`

353 executed in groups. Future versions of Julia might change the algorithm. Note that the
354 elements are not reordered if you use an ordered collection.

358 julia> reduce(*, [2; 3; 4])
359 24
360

479 function mapreduce_impl(f, op::Union{typeof(max), typeof(min)},
480 A::AbstractArray, first::Int, last::Int)
481 a1 = @inbounds A[first]
482 v1 = mapreduce_first(f, op, a1)
483 v2 = v3 = v4 = v1
484 chunk_len = 256
485 start = first + 1
486 simdstop = start + chunk_len - 4
487 while simdstop <= last - 3
488 # short circuit in case of NaN
489 v1 == v1 || return v1
490 v2 == v2 || return v2
491 v3 == v3 || return v3
492 v4 == v4 || return v4
493 @inbounds for i in start:4:simdstop
494 v1 = _fast(op, v1, f(A[i+0]))
495 v2 = _fast(op, v2, f(A[i+1]))
496 v3 = _fast(op, v3, f(A[i+2]))
497 v4 = _fast(op, v4, f(A[i+3]))
498 end
499 checkbounds(A, simdstop+3)
500 start += chunk_len
501 simdstop += chunk_len
502 end
503 v = op(op(v1,v2),op(v3,v4))
504 for i in start:last
505 @inbounds ai = A[i]
506 v = op(v, f(ai))
507 end
508
509 # enforce correct order of 0.0 and -0.0
510 # e.g. maximum([0.0, -0.0]) === 0.0
511 # should hold
512 if isbadzero(op, v)
513 for i in first:last
514 x = @inbounds A[i]
515 isgoodzero(op,x) && return x
516 end
517 end
518 return v
519 end
