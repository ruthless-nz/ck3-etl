33 function repl_cmd(cmd, out)
34 shell = shell_split(get(ENV, "JULIA_SHELL", get(ENV, "SHELL", "/bin/sh")))
35 shell_name = Base.basename(shell[1])
36
37 # Immediately expand all arguments, so that typing e.g. ~/bin/foo works.
38 cmd.exec .= expanduser.(cmd.exec)
39
40 if isempty(cmd.exec)
41 throw(ArgumentError("no cmd to execute"))
42 elseif cmd.exec[1] == "cd"
43 new_oldpwd = pwd()
44 if length(cmd.exec) > 2
45 throw(ArgumentError("cd method only takes one argument"))
46 elseif length(cmd.exec) == 2
47 dir = cmd.exec[2]
48 if dir == "-"
49 if !haskey(ENV, "OLDPWD")
50 error("cd: OLDPWD not set")
51 end
52 cd(ENV["OLDPWD"])
53 else
54 @static if !Sys.iswindows()
55 # TODO: this is a rather expensive way to copy a string, remove?
56 # If it's intended to simulate `cd`, it should instead be doing
57 # more nearly `cd $dir && printf %s \$PWD` (with appropriate quoting),
58 # since shell `cd` does more than just `echo` the result.
59 dir = read(`$shell -c "printf '%s' $(shell_escape_posixly(dir))"`, String)
60 end
61 cd(dir)
62 end
63 else
64 cd()
65 end
66 ENV["OLDPWD"] = new_oldpwd
67 println(out, pwd())
68 else
69 @static if !Sys.iswindows()
70 if shell_name == "fish"
71 shell_escape_cmd = "begin; $(shell_escape_posixly(cmd)); and true; end"
72 else
73 shell_escape_cmd = "($(shell_escape_posixly(cmd))) && true"
74 end
75 cmd = `$shell -c $shell_escape_cmd`
76 end
77 run(ignorestatus(cmd))
78 end
79 nothing

116 function eval_user_input(errio, @nospecialize(ast), show_value::Bool)
117 errcount = 0
118 lasterr = nothing
119 while true
120 try
121 if have_color
122 print(color_normal)
123 end
124 if lasterr !== nothing
125 invokelatest(display_error, errio, lasterr)
126 errcount = 0
127 lasterr = nothing
128 else
129 ast = Meta.lower(Main, ast)
130 value = Core.eval(Main, ast)
131 ccall(:jl_set_global, Cvoid, (Any, Any, Any), Main, :ans, value)
132 if !(value === nothing) && show_value
133 if have_color
134 print(answer_color())
135 end
136 try
137 invokelatest(display, value)
138 catch
139 @error "Evaluation succeeded, but an error occurred while displaying the value" typeof(value)
140 rethrow()
141 end
142 println()
143 end
144 end

146 catch

162 function _parse_input_line_core(s::String, filename::String)
163 ex = ccall(:jl_parse_all, Any, (Ptr{UInt8}, Csize_t, Ptr{UInt8}, Csize_t),
164 s, sizeof(s), filename, sizeof(filename))
165 if ex isa Expr && ex.head === :toplevel
166 if isempty(ex.args)
167 return nothing
168 end
169 last = ex.args[end]
170 if last isa Expr && (last.head === :error || last.head === :incomplete)
171 # if a parse error happens in the middle of a multi-line input
172 # return only the error, so that none of the input is evaluated.
173 return last
174 end
175 end
176 return ex
177 end

192 function parse_input_line(io::IO)
193 s = ""

195 s *= readline(io, keep=true)

203 # detect the reason which caused an :incomplete expression
204 # from the error message
205 # NOTE: the error messages are defined in src/julia-parser.scm
206 incomplete_tag(ex) = :none
207 function incomplete_tag(ex::Expr)
208 Meta.isexpr(ex, :incomplete) || return :none
209 msg = ex.args[1]
210 occursin("string", msg) && return :string
211 occursin("comment", msg) && return :comment
212 occursin("requires end", msg) && return :block
213 occursin("\"`\"", msg) && return :cmd
214 occursin("character", msg) && return :char
215 return :other
216 end

221 function exec_options(opts)

266 startup && load_julia_startup()
267
268 # process cmds list
269 for (cmd, arg) in cmds

271 Core.eval(Main, parse_input_line(arg))
272 elseif cmd == 'E'
273 invokelatest(show, Core.eval(Main, parse_input_line(arg)))
274 println()
275 elseif cmd == 'L'
276 # load file immediately on all processors
277 if !distributed_mode
278 include(Main, arg)
279 else

292 ccall(:jl_exit_on_sigint, Cvoid, (Cint,), 1)

301 end
302 end

317 function load_julia_startup()
318 # If the user built us with a specific Base.SYSCONFDIR, check that location first for a startup.jl file
319 # If it is not found, then continue on to the relative path based on Sys.BINDIR
320 BINDIR = Sys.BINDIR::String
321 SYSCONFDIR = Base.SYSCONFDIR::String
322 if !isempty(SYSCONFDIR) && isfile(joinpath(BINDIR, SYSCONFDIR, "julia", "startup.jl"))
323 include(Main, abspath(BINDIR, SYSCONFDIR, "julia", "startup.jl"))
324 else

359 function run_main_repl(interactive::Bool, quiet::Bool, banner::Bool, history_file::Bool, color_set::Bool)
360 global active_repl
361 # load interactive-only libraries
362 if !isdefined(Main, :InteractiveUtils)
363 try
364 let InteractiveUtils = require(PkgId(UUID(0xb77e0a4c_d291_57a0_90e8_8db25a27a240), "InteractiveUtils"))
365 Core.eval(Main, :(const InteractiveUtils = $InteractiveUtils))
366 Core.eval(Main, :(using .InteractiveUtils))
367 end
368 catch ex
369 @warn "Failed to import InteractiveUtils into module Main" exception=(ex, catch_backtrace())
370 end
371 end
372
373 if interactive && isassigned(REPL_MODULE_REF)
374 invokelatest(REPL_MODULE_REF[]) do REPL
375 term_env = get(ENV, "TERM", @static Sys.iswindows() ? "" : "dumb")
376 term = REPL.Terminals.TTYTerminal(term_env, stdin, stdout, stderr)
377 color_set || (global have_color = REPL.Terminals.hascolor(term))
378 banner && Base.banner(term)
379 if term.term_type == "dumb"
380 active_repl = REPL.BasicREPL(term)
381 quiet || @warn "Terminal not fully functional"
382 else
383 active_repl = REPL.LineEditREPL(term, have_color, true)
384 active_repl.history_file = history_file
385 end
386 # Make sure any displays pushed in .julia/config/startup.jl ends up above the
387 # REPLDisplay
388 pushdisplay(REPL.REPLDisplay(active_repl))
389 _atreplinit(active_repl)
390 REPL.run_repl(active_repl, backend->(global active_repl_backend = backend))
391 end
392 else
393 # otherwise provide a simple fallback
394 if interactive && !quiet
395 @warn "REPL provider not available: using basic fallback"
396 end
397 banner && Base.banner()
398 let input = stdin
399 if isa(input, File) || isa(input, IOStream)
400 # for files, we can slurp in the whole thing at once
401 ex = parse_input_line(read(input, String))
402 if Meta.isexpr(ex, :toplevel)
403 # if we get back a list of statements, eval them sequentially
404 # as if we had parsed them sequentially
405 for stmt in ex.args
406 eval_user_input(stderr, stmt, true)
407 end
408 body = ex.args
409 else
410 eval_user_input(stderr, ex, true)
411 end
412 else
413 while isopen(input) || !eof(input)
414 if interactive
415 print("julia> ")
416 flush(stdout)
417 end
418 try
419 eval_user_input(stderr, parse_input_line(input), true)
420 catch err
421 isa(err, InterruptException) ? print("\n\n") : rethrow()
422 end
423 end
424 end
425 end
426 end
