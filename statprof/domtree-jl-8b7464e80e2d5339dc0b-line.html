206 function snca_compress!(state::Vector{Node}, ancestors::Vector{DFSNumber},
207 v::DFSNumber, last_linked::DFSNumber)
208 u = ancestors[v]
209 @assert u < v
210 if u >= last_linked
211 snca_compress!(state, ancestors, u, last_linked)
212 if state[u].label < state[v].label
213 state[v] = Node(state[v].semi, state[u].label)
214 end

216 end
