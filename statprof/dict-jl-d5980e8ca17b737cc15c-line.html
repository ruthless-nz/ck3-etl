741 function in(key_value::Pair, dict::ImmutableDict, valcmp=(==))
742 key, value = key_value
743 while isdefined(dict, :parent)
744 if dict.key == key
745 valcmp(value, dict.value) && return true
746 end
747 dict = dict.parent
748 end
749 return false
750 end

752 function haskey(dict::ImmutableDict, key)

760 function getindex(dict::ImmutableDict, key)

763 dict = dict.parent
764 end

767 function get(dict::ImmutableDict, key, default)
768 while isdefined(dict, :parent)
769 dict.key == key && return dict.value
770 dict = dict.parent
771 end
772 return default
