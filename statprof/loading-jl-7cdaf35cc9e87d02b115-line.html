19 # in case-sensitive mode, this will still work but will be redundant.

21 # Constants from <sys/attr.h>
22 const ATRATTR_BIT_MAP_COUNT = 5
23 const ATTR_CMN_NAME = 1
24 const BITMAPCOUNT = 1
25 const COMMONATTR = 5
26 const FSOPT_NOFOLLOW = 1 # Don't follow symbolic links

28 const attr_list = zeros(UInt8, 24)
29 attr_list[BITMAPCOUNT] = ATRATTR_BIT_MAP_COUNT
30 attr_list[COMMONATTR] = ATTR_CMN_NAME
31
32 # This essentially corresponds to the following C code:
33 # attrlist attr_list;
34 # memset(&attr_list, 0, sizeof(attr_list));
35 # attr_list.bitmapcount = ATTR_BIT_MAP_COUNT;
36 # attr_list.commonattr = ATTR_CMN_NAME;
37 # struct Buffer {
38 # u_int32_t total_length;
39 # u_int32_t filename_offset;
40 # u_int32_t filename_length;
41 # char filename[max_filename_length];
42 # };
43 # Buffer buf;
44 # getattrpath(path, &attr_list, &buf, sizeof(buf), FSOPT_NOFOLLOW);
45 function isfile_casesensitive(path)
46 isfile(path) || return false
47 path_basename = String(basename(path))
48 local casepreserved_basename
49 header_size = 12
50 buf = Vector{UInt8}(undef, length(path_basename) + header_size + 1)
51 while true
52 ret = ccall(:getattrlist, Cint,
53 (Cstring, Ptr{Cvoid}, Ptr{Cvoid}, Csize_t, Culong),
54 path, attr_list, buf, sizeof(buf), FSOPT_NOFOLLOW)
55 systemerror(:getattrlist, ret ≠ 0)
56 filename_length = GC.@preserve buf unsafe_load(
57 convert(Ptr{UInt32}, pointer(buf) + 8))
58 if (filename_length + header_size) > length(buf)
59 resize!(buf, filename_length + header_size)
60 continue
61 end
62 casepreserved_basename =
63 view(buf, (header_size+1):(header_size+filename_length-1))
64 break
65 end
66 # Hack to compensate for inability to create a string from a subarray with no allocations.
67 codeunits(path_basename) == casepreserved_basename && return true
68
69 # If there is no match, it's possible that the file does exist but HFS+
70 # performed unicode normalization. See https://developer.apple.com/library/mac/qa/qa1235/_index.html.
71 isascii(path_basename) && return false
72 codeunits(Unicode.normalize(path_basename, :NFD)) == casepreserved_basename
73 end

715 function _require_from_serialized(path::String)
716 # loads a precompile cache file, ignoring stale_cachfile tests
717 # load all of the dependent modules first
718 local depmodnames
719 io = open(path, "r")
720 try
721 isvalid_cache_header(io) || return ArgumentError("Invalid header in cache file $path.")
722 depmodnames = parse_cache_header(io)[3]
723 isvalid_file_crc(io) || return ArgumentError("Invalid checksum in cache file $path.")
724 finally
725 close(io)
726 end
727 ndeps = length(depmodnames)
728 depmods = Vector{Any}(undef, ndeps)
729 for i in 1:ndeps
730 modkey, build_id = depmodnames[i]
731 dep = _tryrequire_from_serialized(modkey, build_id, nothing)
732 dep === nothing && return ErrorException("Required dependency $modkey failed to load from a cache file.")
733 depmods[i] = dep::Module
734 end
735 # then load the file
736 return _include_from_serialized(path, depmods)
737 end

739 # returns `true` if require found a precompile cache for this sourcepath, but couldn't load it
740 # returns `false` if the module isn't known to be precompilable
741 # returns the set of modules restored if the cache load succeeded
742 function _require_search_from_serialized(pkg::PkgId, sourcepath::String)
743 paths = find_all_in_cache_path(pkg)
744 for path_to_try in paths::Vector{String}
745 staledeps = stale_cachefile(sourcepath, path_to_try)
746 if staledeps === true
747 continue
748 end
749 # finish loading module graph into staledeps
750 for i in 1:length(staledeps)
751 dep = staledeps[i]
752 dep isa Module && continue
753 modpath, modkey, build_id = dep::Tuple{String, PkgId, UInt64}
754 dep = _tryrequire_from_serialized(modkey, build_id, modpath)
755 if dep === nothing
756 @debug "Required dependency $modkey failed to load from cache file for $modpath."
757 staledeps = true
758 break
759 end
760 staledeps[i] = dep::Module
761 end
762 if staledeps === true
763 continue
764 end
765 restored = _include_from_serialized(path_to_try, staledeps)
766 if isa(restored, Exception)

768 else

770 end
771 end

966 function _require(pkg::PkgId)
967 # handle recursive calls to require
968 loading = get(package_locks, pkg, false)
969 if loading !== false
970 # load already in progress for this module
971 wait(loading)
972 return
973 end
974 package_locks[pkg] = Condition()
975
976 last = toplevel_load[]
977 try
978 toplevel_load[] = false
979 # perform the search operation to select the module file require intends to load
980 path = locate_package(pkg)
981 if path === nothing
982 throw(ArgumentError("""
983 Package $pkg is required but does not seem to be installed:
984 - Run `Pkg.instantiate()` to install all recorded dependencies.
985 """))
986 end
987
988 # attempt to load the module file via the precompile cache locations
989 if JLOptions().use_compiled_modules != 0
990 m = _require_search_from_serialized(pkg, path)
991 if !isa(m, Bool)
992 return
993 end
994 end
995
996 # if the module being required was supposed to have a particular version
997 # but it was not handled by the precompile loader, complain
998 for (concrete_pkg, concrete_build_id) in _concrete_dependencies
999 if pkg == concrete_pkg
1000 @warn """Module $(pkg.name) with build ID $concrete_build_id is missing from the cache.
1001 This may mean $pkg does not support precompilation but is imported by a module that does."""
1002 if JLOptions().incremental != 0
1003 # during incremental precompilation, this should be fail-fast
1004 throw(PrecompilableError())
1005 end
1006 end
1007 end
1008
1009 if JLOptions().use_compiled_modules != 0
1010 if (0 == ccall(:jl_generating_output, Cint, ())) || (JLOptions().incremental != 0)
1011 # spawn off a new incremental pre-compile task for recursive `require` calls
1012 # or if the require search declared it was pre-compiled before (and therefore is expected to still be pre-compilable)
1013 cachefile = compilecache(pkg, path)
1014 if isa(cachefile, Exception)
1015 if precompilableerror(cachefile)
1016 verbosity = isinteractive() ? CoreLogging.Info : CoreLogging.Debug
1017 @logmsg verbosity "Skipping precompilation since __precompile__(false). Importing $pkg."
1018 else
1019 @warn "The call to compilecache failed to create a usable precompiled cache file for $pkg" exception=m
1020 end
1021 # fall-through to loading the file locally
1022 else
1023 m = _require_from_serialized(cachefile)
1024 if isa(m, Exception)
1025 @warn "The call to compilecache failed to create a usable precompiled cache file for $pkg" exception=m
1026 else
1027 return
1028 end
1029 end
1030 end
1031 end
1032
1033 # just load the file normally via include
1034 # for unknown dependencies
1035 uuid = pkg.uuid
1036 uuid = (uuid === nothing ? (UInt64(0), UInt64(0)) : convert(NTuple{2, UInt64}, uuid))
1037 old_uuid = ccall(:jl_module_uuid, NTuple{2, UInt64}, (Any,), __toplevel__)
1038 if uuid !== old_uuid
1039 ccall(:jl_set_module_uuid, Cvoid, (Any, NTuple{2, UInt64}), __toplevel__, uuid)
1040 end
1041 try
1042 include_relative(__toplevel__, path)
1043 return
1044 finally
1045 if uuid !== old_uuid
1046 ccall(:jl_set_module_uuid, Cvoid, (Any, NTuple{2, UInt64}), __toplevel__, old_uuid)
1047 end
1048 end
1049 finally
1050 toplevel_load[] = last
1051 loading = pop!(package_locks, pkg)
1052 notify(loading, all=true)
1053 end
1054 nothing
1055 end

1218 for important notes.
1219 """
1220 function compilecache(pkg::PkgId)
1221 path = locate_package(pkg)
1222 path === nothing && throw(ArgumentError("$pkg not found during precompilation"))
1223 return compilecache(pkg, path)
1224 end
1225 function compilecache(pkg::PkgId, path::String)
1226 # decide where to put the resulting cache file
1227 cachefile = abspath(DEPOT_PATH[1], cache_file_entry(pkg))
1228 cachepath = dirname(cachefile)
1229 isdir(cachepath) || mkpath(cachepath)
1230 # build up the list of modules that we want the precompile process to preserve
1231 concrete_deps = copy(_concrete_dependencies)
1232 for (key, mod) in loaded_modules
1233 if !(mod === Main || mod === Core || mod === Base)
1234 push!(concrete_deps, key => module_build_id(mod))
1235 end
1236 end
1237 # run the expression and cache the result
1238 verbosity = isinteractive() ? CoreLogging.Info : CoreLogging.Debug
1239 if isfile(cachefile)
1240 @logmsg verbosity "Recompiling stale cache file $cachefile for $pkg"
1241 else
1242 @logmsg verbosity "Precompiling $pkg"
1243 end
1244 p = create_expr_cache(path, cachefile, concrete_deps, pkg.uuid)
1245 if success(p)
1246 # append checksum to the end of the .ji file:
1247 open(cachefile, "a+") do f
1248 write(f, _crc32c(seekstart(f)))
1249 end
1250 elseif p.exitcode == 125
1251 return PrecompilableError()
1252 else
1253 error("Failed to precompile $pkg to $cachefile.")
1254 end
1255 return cachefile
1256 end

1263 function parse_cache_header(f::IO)
1264 modules = Vector{Pair{PkgId, UInt64}}()
1265 while true
1266 n = read(f, Int32)
1267 n == 0 && break
1268 sym = String(read(f, n)) # module name
1269 uuid = UUID((read(f, UInt64), read(f, UInt64))) # pkg UUID
1270 build_id = read(f, UInt64) # build UUID (mostly just a timestamp)
1271 push!(modules, PkgId(uuid, sym) => build_id)
1272 end
1273 totbytes = read(f, Int64) # total bytes for file dependencies
1274 # read the list of requirements
1275 # and split the list into include and requires statements
1276 includes = Tuple{PkgId, String, Float64}[]
1277 requires = Pair{PkgId, PkgId}[]
1278 while true
1279 n2 = read(f, Int32)
1280 n2 == 0 && break
1281 depname = String(read(f, n2))
1282 mtime = read(f, Float64)
1283 n1 = read(f, Int32)
1284 # map ids to keys
1285 modkey = (n1 == 0) ? PkgId("") : modules[n1].first
1286 if n1 != 0
1287 # consume (and ignore) the module path too
1288 while true
1289 n1 = read(f, Int32)
1290 totbytes -= 4
1291 n1 == 0 && break
1292 skip(f, n1) # String(read(f, n1))
1293 totbytes -= n1
1294 end
1295 end
1296 if depname[1] == '\0'
1297 push!(requires, modkey => binunpack(depname))
1298 else
1299 push!(includes, (modkey, depname, mtime))
1300 end
1301 totbytes -= 4 + 4 + n2 + 8
1302 end
1303 @assert totbytes == 12 "header of cache file appears to be corrupt"
1304 srctextpos = read(f, Int64)
1305 # read the list of modules that are required to be present during loading
1306 required_modules = Vector{Pair{PkgId, UInt64}}()
1307 while true
1308 n = read(f, Int32)
1309 n == 0 && break
1310 sym = String(read(f, n)) # module name
1311 uuid = UUID((read(f, UInt64), read(f, UInt64))) # pkg UUID
1312 build_id = read(f, UInt64) # build id
1313 push!(required_modules, PkgId(uuid, sym) => build_id)
1314 end
1315 return modules, (includes, requires), required_modules, srctextpos
1316 end

1350 function _read_dependency_src(io::IO, filename::AbstractString)
1351 while !eof(io)
1352 filenamelen = read(io, Int32)
1353 filenamelen == 0 && break

1355 len = read(io, UInt64)
1356 if fn == filename
1357 return String(read(io, len))

1374 # returns true if it "cachefile.ji" is stale relative to "modpath.jl"
1375 # otherwise returns the list of dependencies to also check
1376 function stale_cachefile(modpath::String, cachefile::String)
1377 io = open(cachefile, "r")
1378 try
1379 if !isvalid_cache_header(io)
1380 @debug "Rejecting cache file $cachefile due to it containing an invalid cache header"
1381 return true # invalid cache file
1382 end
1383 (modules, (includes, requires), required_modules) = parse_cache_header(io)
1384 modules = Dict{PkgId, UInt64}(modules)
1385
1386 # Check if transitive dependencies can be fulfilled
1387 ndeps = length(required_modules)
1388 depmods = Vector{Any}(undef, ndeps)
1389 for i in 1:ndeps
1390 req_key, req_build_id = required_modules[i]
1391 # Module is already loaded
1392 if root_module_exists(req_key)
1393 M = root_module(req_key)
1394 if PkgId(M) == req_key && module_build_id(M) === req_build_id
1395 depmods[i] = M
1396 else
1397 @debug "Rejecting cache file $cachefile because module $req_key is already loaded and incompatible."
1398 return true # Won't be able to fulfill dependency
1399 end
1400 else
1401 path = locate_package(req_key)
1402 if path === nothing
1403 @debug "Rejecting cache file $cachefile because dependency $req_key not found."
1404 return true # Won't be able to fulfill dependency
1405 end
1406 depmods[i] = (path, req_key, req_build_id)
1407 end
1408 end
1409
1410 # check if this file is going to provide one of our concrete dependencies
1411 # or if it provides a version that conflicts with our concrete dependencies
1412 # or neither
1413 skip_timecheck = false
1414 for (req_key, req_build_id) in _concrete_dependencies
1415 build_id = get(modules, req_key, UInt64(0))
1416 if build_id !== UInt64(0)
1417 if build_id === req_build_id
1418 skip_timecheck = true
1419 break
1420 end
1421 @debug "Rejecting cache file $cachefile because it provides the wrong uuid (got $build_id) for $mod (want $req_build_id)"
1422 return true # cachefile doesn't provide the required version of the dependency
1423 end
1424 end
1425
1426 # now check if this file is fresh relative to its source files
1427 if !skip_timecheck
1428 if !samefile(includes[1][2], modpath)
1429 @debug "Rejecting cache file $cachefile because it is for file $(includes[1][2])) not file $modpath"
1430 return true # cache file was compiled from a different path
1431 end
1432 for (modkey, req_modkey) in requires
1433 # verify that `require(modkey, name(req_modkey))` ==> `req_modkey`
1434 if identify_package(modkey, req_modkey.name) != req_modkey
1435 @debug "Rejecting cache file $cachefile because uuid mapping for $modkey => $req_modkey has changed"
1436 return true
1437 end
1438 end
1439 for (_, f, ftime_req) in includes
1440 # Issue #13606: compensate for Docker images rounding mtimes
1441 # Issue #20837: compensate for GlusterFS truncating mtimes to microseconds
1442 ftime = mtime(f)
1443 if ftime != ftime_req && ftime != floor(ftime_req) && ftime != trunc(ftime_req, digits=6)
1444 @debug "Rejecting stale cache file $cachefile (mtime $ftime_req) because file $f (mtime $ftime) has changed"
1445 return true
1446 end
1447 end
1448 end
1449
1450 if !isvalid_file_crc(io)
1451 @debug "Rejecting cache file $cachefile because it has an invalid checksum"
1452 return true
1453 end
1454
1455 return depmods # fresh cachefile
1456 finally
1457 close(io)
1458 end
1459 end
