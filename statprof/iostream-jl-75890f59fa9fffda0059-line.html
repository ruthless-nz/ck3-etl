220 """
221 open_flags(; keywords...) -> NamedTuple
222
223 Compute the `read`, `write`, `create`, `truncate`, `append` flag value for
224 a given set of keyword arguments to [`open`](@ref) a [`NamedTuple`](@ref).
225 """
226 function open_flags(;
227 read :: Union{Bool,Nothing} = nothing,
228 write :: Union{Bool,Nothing} = nothing,
229 create :: Union{Bool,Nothing} = nothing,
230 truncate :: Union{Bool,Nothing} = nothing,
231 append :: Union{Bool,Nothing} = nothing,
232 )
233 if write === true && read !== true && append !== true
234 create === nothing && (create = true)
235 truncate === nothing && (truncate = true)
236 end
237
238 if truncate === true || append === true
239 write === nothing && (write = true)
240 create === nothing && (create = true)
241 end
242
243 write === nothing && (write = false)
244 read === nothing && (read = !write)
245 create === nothing && (create = false)
246 truncate === nothing && (truncate = false)
247 append === nothing && (append = false)
248
249 return (
250 read = read,
251 write = write,
252 create = create,
253 truncate = truncate,
254 append = append,
255 )
256 end

266 | `write` | open for writing | `truncate \\| append` |
267 | `create` | create if non-existent | `!read & write \\| truncate \\| append` |
268 | `truncate` | truncate to zero size | `!read & write` |
269 | `append` | seek to end | `false` |
