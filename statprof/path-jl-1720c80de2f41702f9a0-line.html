66 buf = Vector{UInt8}(undef, path_max)
67 sz = RefValue{Csize_t}(path_max + 1)

69 rc = ccall(:uv_os_homedir, Cint, (Ptr{UInt8}, Ptr{Csize_t}), buf, sz)

210 """
211 splitpath(path::AbstractString) -> Vector{String}
212
213 Split a file path into all its path components. This is the opposite of
214 `joinpath`. Returns an array of substrings, one for each directory or file in
215 the path, including the root directory if present.
216
217 !!! compat "Julia 1.1"
218 This function requires at least Julia 1.1.
219
220 # Examples
221 ```jldoctest
222 julia> splitpath("/home/myuser/example.jl")
223 4-element Array{String,1}:
224 "/"
225 "home"
226 "myuser"
227 "example.jl"
228 ```
229 """
230 function splitpath(p::String)
231 drive, p = splitdrive(p)
232 out = String[]
233 isempty(p) && (pushfirst!(out,p)) # "" means the current directory.
234 while !isempty(p)
235 dir, base = _splitdir_nodrive(p)
236 dir == p && (pushfirst!(out, dir); break) # Reached root node.
237 if !isempty(base) # Skip trailing '/' in basename
238 pushfirst!(out, base)
239 end
240 p = dir
241 end
242 if !isempty(drive) # Tack the drive back on to the first element.
243 out[1] = drive*out[1] # Note that length(out) is always >= 1.
244 end
245 return out
246 end

265 function joinpath(a::String, b::String)
266 isabspath(b) && return b
267 A, a = splitdrive(a)
268 B, b = splitdrive(b)
269 !isempty(B) && A != B && return string(B,b)
270 C = isempty(B) ? A : B
271 isempty(a) ? string(C,b) :

277 """
278 normpath(path::AbstractString) -> AbstractString
279
280 Normalize a path, removing "." and ".." entries.
281
282 # Examples
283 ```jldoctest
284 julia> normpath("/home/myuser/../example.jl")
285 "/home/example.jl"
286 ```
287 """
288 function normpath(path::String)
289 isabs = isabspath(path)
290 isdir = isdirpath(path)
291 drive, path = splitdrive(path)
292 parts = split(path, path_separator_re)
293 filter!(x->!isempty(x) && x!=".", parts)
294 while true
295 clean = true
296 for j = 1:length(parts)-1
297 if parts[j] != ".." && parts[j+1] == ".."
298 deleteat!(parts, j:j+1)
299 clean = false
300 break
301 end
302 end
303 clean && break
304 end
305 if isabs
306 while !isempty(parts) && parts[1] == ".."
307 popfirst!(parts)
308 end
309 elseif isempty(parts)
310 push!(parts, ".")
311 end
312 path = join(parts, path_separator)
313 if isabs
314 path = path_separator*path
315 end
316 if isdir && !isdirpath(path)
317 path *= path_separator
318 end
319 string(drive,path)
320 end
321 normpath(a::AbstractString, b::AbstractString...) = normpath(joinpath(a,b...))

443 """
444 relpath(path::AbstractString, startpath::AbstractString = ".") -> AbstractString
445
446 Return a relative filepath to `path` either from the current directory or from an optional
447 start directory. This is a path computation: the filesystem is not accessed to confirm the
448 existence or nature of `path` or `startpath`.
449 """
450 function relpath(path::String, startpath::String = ".")
451 isempty(path) && throw(ArgumentError("`path` must be specified"))
452 isempty(startpath) && throw(ArgumentError("`startpath` must be specified"))
453 curdir = "."
454 pardir = ".."
455 path == startpath && return curdir
456 path_arr = split(abspath(path), path_separator_re)
457 start_arr = split(abspath(startpath), path_separator_re)
458 i = 0
459 while i < min(length(path_arr), length(start_arr))
460 i += 1
461 if path_arr[i] != start_arr[i]
462 i -= 1
463 break
464 end
465 end
466 pathpart = join(path_arr[i+1:something(findlast(x -> !isempty(x), path_arr), 0)], path_separator)
467 prefix_num = something(findlast(x -> !isempty(x), start_arr), 0) - i - 1
468 if prefix_num >= 0
469 prefix = pardir * path_separator
470 relpath_ = isempty(pathpart) ?
471 (prefix^prefix_num) * pardir :
472 (prefix^prefix_num) * pardir * path_separator * pathpart
473 else
474 relpath_ = pathpart
475 end
476 return isempty(relpath_) ? curdir : relpath_
477 end
