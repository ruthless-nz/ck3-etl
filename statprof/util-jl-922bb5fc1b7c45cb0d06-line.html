41 function GC_Diff(new::GC_Num, old::GC_Num)
42 # logic from `src/gc.c:jl_gc_total_bytes`
43 old_allocd = gc_total_bytes(old)
44 new_allocd = gc_total_bytes(new)
45 return GC_Diff(new_allocd - old_allocd,
46 Int64(new.malloc - old.malloc),
47 Int64(new.realloc - old.realloc),
48 Int64(new.poolalloc - old.poolalloc),
49 Int64(new.bigalloc - old.bigalloc),
50 Int64(new.freecall - old.freecall),
51 Int64(new.total_time - old.total_time),
52 new.pause - old.pause,
53 new.full_sweep - old.full_sweep)
54 end

218 # measure bytes allocated without *most* contamination from compilation
219 # Note: This reports a different value from the @time macros, because
220 # it wraps the call in a function, however, this means that things
221 # like: @allocated y = foo()
222 # will not work correctly, because it will set y in the context of
223 # the local function made by the macro, not the current function
224 """
225 @allocated
226
227 A macro to evaluate an expression, discarding the resulting value, instead returning the
228 total number of bytes allocated during evaluation of the expression. Note: the expression is
229 evaluated inside a local function, instead of the current context, in order to eliminate the
230 effects of compilation, however, there still may be some allocations due to JIT compilation.
231 This also makes the results inconsistent with the `@time` macros, which do not try to adjust
232 for the effects of compilation.
233
234 See also [`@time`](@ref), [`@timev`](@ref), [`@timed`](@ref),
235 and [`@elapsed`](@ref).
236
237 ```julia-repl
238 julia> @allocated rand(10^6)
239 8000080
240 ```
241 """
242 macro allocated(ex)
243 quote
244 let
245 local f
246 function f()
247 b0 = gc_bytes()
248 $(esc(ex))
249 gc_bytes() - b0
250 end
251 f()
252 end
253 end

591 (UInt32, Cwstring, Cwstring, Ptr{UInt8}, Ptr{UInt32}),
592 CRED_PACK_GENERIC_CREDENTIALS, default_username, "", credbuf, credbufsize)
593 @assert succeeded
594
595 # Step 2: Create the actual dialog
596 # 2.1: Set up the window
597 messageArr = Base.cwstring(message)
598 captionArr = Base.cwstring(caption)
599 pfSave = Ref{Bool}(false)
600 cred = Ref{CREDUI_INFO}(CREDUI_INFO(sizeof(CREDUI_INFO), C_NULL, pointer(messageArr), pointer(captionArr), C_NULL))
601 dwflags = CREDUIWIN_GENERIC | CREDUIWIN_ENUMERATE_CURRENT_USER
602 if !prompt_username
603 # Disable setting anything other than default_username
604 dwflags |= CREDUIWIN_IN_CRED_ONLY
605 end
606 authPackage = Ref{Culong}(0)
607 outbuf_data = Ref{Ptr{Cvoid}}(C_NULL)
608 outbuf_size = Ref{Culong}(0)
609
610 # 2.2: Do the actual request
611 code = ccall((:CredUIPromptForWindowsCredentialsW, "credui.dll"), stdcall, UInt32, (Ptr{CREDUI_INFO}, UInt32, Ptr{Culong},
612 Ptr{Cvoid}, Culong, Ptr{Ptr{Cvoid}}, Ptr{Culong}, Ptr{Bool}, UInt32), cred, 0, authPackage, credbuf, credbufsize[],
613 outbuf_data, outbuf_size, pfSave, dwflags)
614
615 # 2.3: If that failed for any reason other than the user canceling, error out.
616 # If the user canceled, just return nothing
617 code == ERROR_CANCELLED && return nothing
618 windowserror(:winprompt, code != ERROR_SUCCESS)

628 (UInt32, Ptr{Cvoid}, UInt32, Ptr{UInt16}, Ptr{UInt32}, Ptr{UInt16}, Ptr{UInt32}, Ptr{UInt16}, Ptr{UInt32}),

649 unsafe_crc32c(a, n, crc) = ccall(:jl_crc32c, UInt32, (UInt32, Ptr{UInt8}, Csize_t), crc, a, n)

651 _crc32c(a::Union{Array{UInt8},FastContiguousSubArray{UInt8,N,<:Array{UInt8}} where N}, crc::UInt32=0x00000000) =
652 unsafe_crc32c(a, length(a) % Csize_t, crc)
653
654 _crc32c(s::String, crc::UInt32=0x00000000) = unsafe_crc32c(s, sizeof(s) % Csize_t, crc)
655
656 function _crc32c(io::IO, nb::Integer, crc::UInt32=0x00000000)
657 nb < 0 && throw(ArgumentError("number of bytes to checksum must be 0, got $nb"))
658 # use block size 24576=8192*3, since that is the threshold for
659 # 3-way parallel SIMD code in the underlying jl_crc32c C function.
660 buf = Vector{UInt8}(undef, min(nb, 24576))
661 while !eof(io) && nb > 24576
662 n = readbytes!(io, buf)
663 crc = unsafe_crc32c(buf, n, crc)
664 nb -= n
665 end
666 return unsafe_crc32c(buf, readbytes!(io, buf, min(nb, length(buf))), crc)
667 end

669 _crc32c(io::IOStream, crc::UInt32=0x00000000) = _crc32c(io, filesize(io)-position(io), crc)
670 _crc32c(uuid::UUID, crc::UInt32=0x00000000) =
671 ccall(:jl_crc32c, UInt32, (UInt32, Ref{UInt128}, Csize_t), crc, uuid.value, 16)

673 """
674 @kwdef typedef
675
676 This is a helper macro that automatically defines a keyword-based constructor for the type
677 declared in the expression `typedef`, which must be a `struct` or `mutable struct`
678 expression. The default argument is supplied by declaring fields of the form `field::T =
679 default` or `field = default`. If no default is provided then the keyword argument becomes
680 a required keyword argument in the resulting type constructor.
681
682 Inner constructors can still be defined, but at least one should accept arguments in the
683 same form as the default inner constructor (i.e. one positional argument per field) in
684 order to function correctly with the keyword outer constructor.
685
686 !!! compat "Julia 1.1"
687 `Base.@kwdef` for parametric structs, and structs with supertypes
688 requires at least Julia 1.1.
689
690 # Examples
691 ```jldoctest
692 julia> Base.@kwdef struct Foo
693 a::Int = 1 # specified default
694 b::String # required keyword
695 end
696 Foo
697
698 julia> Foo(b="hi")
699 Foo(1, "hi")
700
701 julia> Foo()
702 ERROR: UndefKeywordError: keyword argument b not assigned
703 Stacktrace:
704 [...]
705 ```
706 """
707 macro kwdef(expr)
708 expr = macroexpand(__module__, expr) # to expand @static
709 expr isa Expr && expr.head == :struct || error("Invalid usage of @kwdef")
710 T = expr.args[2]
711 if T isa Expr && T.head == :<:
712 T = T.args[1]
713 end
714
715 params_ex = Expr(:parameters)
716 call_args = Any[]
717
718 _kwdef!(expr.args[3], params_ex.args, call_args)
719 # Only define a constructor if the type has fields, otherwise we'll get a stack
720 # overflow on construction
721 if !isempty(params_ex.args)
722 if T isa Symbol
723 kwdefs = :(($(esc(T)))($params_ex) = ($(esc(T)))($(call_args...)))
724 elseif T isa Expr && T.head == :curly
725 # if T == S{A<:AA,B<:BB}, define two methods
726 # S(...) = ...
727 # S{A,B}(...) where {A<:AA,B<:BB} = ...
728 S = T.args[1]
729 P = T.args[2:end]
730 Q = [U isa Expr && U.head == :<: ? U.args[1] : U for U in P]
731 SQ = :($S{$(Q...)})
732 kwdefs = quote
733 ($(esc(S)))($params_ex) =($(esc(S)))($(call_args...))
734 ($(esc(SQ)))($params_ex) where {$(esc.(P)...)} =
735 ($(esc(SQ)))($(call_args...))
736 end
737 else
738 error("Invalid usage of @kwdef")
739 end
740 else
741 kwdefs = nothing
742 end
743 quote
744 Base.@__doc__($(esc(expr)))
745 $kwdefs
746 end
747 end
748
749 # @kwdef helper function
750 # mutates arguments inplace
751 function _kwdef!(blk, params_args, call_args)
752 for i in eachindex(blk.args)
753 ei = blk.args[i]
754 if ei isa Symbol
755 # var
756 push!(params_args, ei)
757 push!(call_args, ei)
758 elseif ei isa Expr
759 if ei.head == :(=)
760 lhs = ei.args[1]
761 if lhs isa Symbol
762 # var = defexpr
763 var = lhs
764 elseif lhs isa Expr && lhs.head == :(::) && lhs.args[1] isa Symbol
765 # var::T = defexpr
766 var = lhs.args[1]
767 else
768 # something else, e.g. inline inner constructor
769 # F(...) = ...
770 continue
771 end
772 defexpr = ei.args[2] # defexpr
773 push!(params_args, Expr(:kw, var, esc(defexpr)))
774 push!(call_args, var)
775 blk.args[i] = lhs
776 elseif ei.head == :(::) && ei.args[1] isa Symbol
777 # var::Typ
778 var = ei.args[1]
779 push!(params_args, var)
780 push!(call_args, var)
781 elseif ei.head == :block
782 # can arise with use of @static inside type decl
783 _kwdef!(ei, params_args, call_args)
784 end
785 end
786 end
787 blk
788 end

792 """
793 Base.runtests(tests=["all"]; ncores=ceil(Int, Sys.CPU_THREADS / 2),
794 exit_on_error=false, [seed])
795
796 Run the Julia unit tests listed in `tests`, which can be either a string or an array of
797 strings, using `ncores` processors. If `exit_on_error` is `false`, when one test
798 fails, all remaining tests in other files will still be run; they are otherwise discarded,
799 when `exit_on_error == true`.
800 If a seed is provided via the keyword argument, it is used to seed the
801 global RNG in the context where the tests are run; otherwise the seed is chosen randomly.
802 """
803 function runtests(tests = ["all"]; ncores = ceil(Int, Sys.CPU_THREADS / 2),
804 exit_on_error=false,
805 seed::Union{BitInteger,Nothing}=nothing)
806 if isa(tests,AbstractString)
807 tests = split(tests)
808 end
809 exit_on_error && push!(tests, "--exit-on-error")
810 seed !== nothing && push!(tests, "--seed=0x$(string(seed % UInt128, base=16))") # cast to UInt128 to avoid a minus sign
811 ENV2 = copy(ENV)
812 ENV2["JULIA_CPU_THREADS"] = "$ncores"
813 try
814 run(setenv(`$(julia_cmd()) $(joinpath(Sys.BINDIR::String,
815 Base.DATAROOTDIR, "julia", "test", "runtests.jl")) $tests`, ENV2))
816 catch
817 buf = PipeBuffer()
818 Base.require(Base, :InteractiveUtils).versioninfo(buf)
819 error("A test has failed. Please submit a bug report (https://github.com/JuliaLang/julia/issues)\n" *
820 "including error messages above and the output of versioninfo():\n$(read(buf, String))")
821 end
