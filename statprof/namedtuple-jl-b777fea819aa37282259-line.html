3 """
4 NamedTuple
5
6 `NamedTuple`s are, as their name suggests, named [`Tuple`](@ref)s. That is, they're a
7 tuple-like collection of values, where each entry has a unique name, represented as a
8 [`Symbol`](@ref). Like `Tuple`s, `NamedTuple`s are immutable; neither the names nor the values
9 can be modified in place after construction.
10
11 Accessing the value associated with a name in a named tuple can be done using field
12 access syntax, e.g. `x.a`, or using [`getindex`](@ref), e.g. `x[:a]`. A tuple of the
13 names can be obtained using [`keys`](@ref), and a tuple of the values can be obtained
14 using [`values`](@ref).
15
16 !!! note
17 Iteration over `NamedTuple`s produces the *values* without the names. (See example
18 below.) To iterate over the name-value pairs, use the [`pairs`](@ref) function.
19
20 # Examples
21 ```jldoctest
22 julia> x = (a=1, b=2)
23 (a = 1, b = 2)
24
25 julia> x.a
26 1
27
28 julia> x[:a]
29 1
30
31 julia> keys(x)
32 (:a, :b)
33
34 julia> values(x)
35 (1, 2)
36
37 julia> collect(x)
38 2-element Array{Int64,1}:
39 1
40 2
41
42 julia> collect(pairs(x))
43 2-element Array{Pair{Symbol,Int64},1}:
44 :a => 1
45 :b => 2
46 ```

190 """
191 merge(a::NamedTuple, bs::NamedTuple...)
192
193 Construct a new named tuple by merging two or more existing ones, in a left-associative
194 manner. Merging proceeds left-to-right, between pairs of named tuples, and so the order of fields
195 present in both the leftmost and rightmost named tuples take the same position as they are found in the
196 leftmost named tuple. However, values are taken from matching fields in the rightmost named tuple that
197 contains that field. Fields present in only the rightmost named tuple of a pair are appended at the end.
198 A fallback is implemented for when only a single named tuple is supplied,
199 with signature `merge(a::NamedTuple)`.
200
201 !!! compat "Julia 1.1"
202 Merging 3 or more `NamedTuple` requires at least Julia 1.1.
203
204 # Examples
205 ```jldoctest
206 julia> merge((a=1, b=2, c=3), (b=4, d=5))
207 (a = 1, b = 4, c = 3, d = 5)
208 ```
209
210 ```jldoctest
211 julia> merge((a=1, b=2), (b=3, c=(d=1,)), (c=(d=2,),))
212 (a = 1, b = 3, c = (d = 2,))
213 ```
214 """
215 function merge(a::NamedTuple{an}, b::NamedTuple{bn}) where {an, bn}
216 if @generated
217 names = merge_names(an, bn)
218 types = merge_types(names, a, b)
219 vals = Any[ :(getfield($(sym_in(n, bn) ? :b : :a), $(QuoteNode(n)))) for n in names ]
220 :( NamedTuple{$names,$types}(($(vals...),)) )
221 else
222 names = merge_names(an, bn)
223 types = merge_types(names, typeof(a), typeof(b))
224 NamedTuple{names,types}(map(n->getfield(sym_in(n, bn) ? b : a, n), names))
225 end
226 end

236 """
237 merge(a::NamedTuple, iterable)
238
239 Interpret an iterable of key-value pairs as a named tuple, and perform a merge.
240
241 ```jldoctest
242 julia> merge((a=1, b=2, c=3), [:b=>4, :d=>5])
243 (a = 1, b = 4, c = 3, d = 5)
244 ```
245 """
246 function merge(a::NamedTuple, itr)
247 names = Symbol[]
248 vals = Any[]
249 inds = IdDict()
250 for (k,v) in itr
251 oldind = get(inds, k, 0)
252 if oldind > 0
253 vals[oldind] = v
254 else
255 push!(names, k)
256 push!(vals, v)
257 inds[k] = length(names)
258 end
259 end
260 merge(a, NamedTuple{(names...,)}((vals...,)))
261 end
