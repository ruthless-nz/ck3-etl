735 ## Abstract Syntax Tree (AST) printing ##
736
737 # Summary:
738 # print(io, ex) defers to show_unquoted(io, ex)
739 # show(io, ex) defers to show_unquoted(io, QuoteNode(ex))
740 # show_unquoted(io, ex) does the heavy lifting
741 #
742 # AST printing should follow two rules:
743 # 1. Meta.parse(string(ex)) == ex
744 # 2. eval(Meta.parse(repr(ex))) == ex
745 #
746 # Rule 1 means that printing an expression should generate Julia code which
747 # could be reparsed to obtain the original expression. This code should be
748 # unambiguous and as readable as possible.
749 #
750 # Rule 2 means that showing an expression should generate a quoted version of
751 # print’s output. Parsing and then evaling this output should return the
752 # original expression.
753 #
754 # This is consistent with many other show methods, i.e.:
755 # show(Set([1,2,3])) # ==> "Set{Int64}([2,3,1])"
756 # eval(Meta.parse("Set{Int64}([2,3,1])”) # ==> An actual set
757 # While this isn’t true of ALL show methods, it is of all ASTs.
758
759 const ExprNode = Union{Expr, QuoteNode, Slot, LineNumberNode, SSAValue,
760 GotoNode, GlobalRef, PhiNode, PhiCNode, UpsilonNode,
761 Core.Compiler.GotoIfNot, Core.Compiler.ReturnNode}
762 # Operators have precedence levels from 1-N, and show_unquoted defaults to a
763 # precedence level of 0 (the fourth argument). The top-level print and show
764 # methods use a precedence of -1 to specially allow space-separated macro syntax
765 print( io::IO, ex::ExprNode) = (show_unquoted(io, ex, 0, -1); nothing)
766 show( io::IO, ex::ExprNode) = show_unquoted_quote_expr(io, ex, 0, -1)

1217 print(io, "(")
1218 show_unquoted(io, func, indent)
1219 print(io, ")")
1220 show_enclosed_list(io, op, func_args, ", ", cl, indent)
1221 else
1222 show_call(io, head, func, func_args, indent)
1223 end
1224
1225 # normal function (i.e. "f(x,y)")
1226 else
1227 show_call(io, head, func, func_args, indent)
1228 end
1229
1230 # new expr
1231 elseif head === :new || head === :splatnew

1233
1234 # other call-like expressions ("A[1,2]", "T{X,Y}", "f.(X,Y)")
1235 elseif haskey(expr_calls, head) && nargs >= 1 # :ref/:curly/:calldecl/:(.)

1237 show_call(io, head, args[1], funcargslike, indent)

1239 # comprehensions
1240 elseif head === :typed_comprehension && nargs == 2
1241 show_unquoted(io, args[1], indent)
1242 print(io, '[')
1243 show_generator(io, args[2], indent)
1244 print(io, ']')
1245
1246 elseif head === :comprehension && nargs == 1
1247 print(io, '[')
1248 show_generator(io, args[1], indent)
1249 print(io, ']')
1250
1251 elseif (head === :generator && nargs >= 2) || (head === :flatten && nargs == 1)
1252 print(io, '(')
1253 show_generator(io, ex, indent)
1254 print(io, ')')
1255
1256 elseif head === :filter && nargs == 2
1257 show_unquoted(io, args[2], indent)
1258 print(io, " if ")
1259 show_unquoted(io, args[1], indent)
1260
1261 # comparison (i.e. "x < y < z")
1262 elseif head === :comparison && nargs >= 3 && (nargs&1==1)
1263 comp_prec = minimum(operator_precedence, args[2:2:end])
1264 if comp_prec <= prec
1265 show_enclosed_list(io, '(', args, " ", ')', indent, comp_prec)
1266 else
1267 show_list(io, args, " ", indent, comp_prec)
1268 end
1269
1270 # function calls need to transform the function from :call to :calldecl
1271 # so that operators are printed correctly
1272 elseif head === :function && nargs==2 && is_expr(args[1], :call)
1273 show_block(io, head, Expr(:calldecl, args[1].args...), args[2], indent)
1274 print(io, "end")
1275
1276 elseif head === :function && nargs == 1
1277 print(io, "function ", args[1], " end")
1278
1279 elseif head === :do && nargs == 2
1280 show_unquoted(io, args[1], indent, -1)
1281 print(io, " do ")
1282 show_list(io, args[2].args[1].args, ", ", 0)
1283 for stmt in args[2].args[2].args
1284 print(io, '\n', " "^(indent + indent_width))
1285 show_unquoted(io, stmt, indent + indent_width, -1)
1286 end
1287 print(io, '\n', " "^indent)
1288 print(io, "end")
1289
1290 # block with argument
1291 elseif head in (:for,:while,:function,:if,:elseif,:let) && nargs==2
1292 show_block(io, head, args[1], args[2], indent)
1293 print(io, "end")
1294
1295 elseif (head === :if || head === :elseif) && nargs == 3
1296 show_block(io, head, args[1], args[2], indent)
1297 if isa(args[3],Expr) && args[3].head == :elseif
1298 show_unquoted(io, args[3], indent, prec)
1299 else
1300 show_block(io, "else", args[3], indent)
1301 print(io, "end")
1302 end
1303
1304 elseif head === :module && nargs==3 && isa(args[1],Bool)
1305 show_block(io, args[1] ? :module : :baremodule, args[2], args[3], indent)
1306 print(io, "end")
1307
1308 # type declaration
1309 elseif head === :struct && nargs==3
1310 show_block(io, args[1] ? Symbol("mutable struct") : Symbol("struct"), args[2], args[3], indent)
1311 print(io, "end")
1312
1313 elseif head === :primitive && nargs == 2
1314 print(io, "primitive type ")
1315 show_list(io, args, ' ', indent)
1316 print(io, " end")
1317
1318 elseif head === :abstract && nargs == 1
1319 print(io, "abstract type ")
1320 show_list(io, args, ' ', indent)
1321 print(io, " end")
1322
1323 # empty return (i.e. "function f() return end")
1324 elseif head === :return && nargs == 1 && args[1] === nothing
1325 print(io, head)
1326

1331

1333 # (i.e. "function f(L...) end" or "f(B...)")
1334 elseif head === :(...) && nargs == 1
1335 show_unquoted(io, args[1], indent)
1336 print(io, "...")
1337
1338 elseif (nargs == 0 && head in (:break, :continue))
1339 print(io, head)
1340
1341 elseif (nargs == 1 && head in (:return, :const)) ||
1342 head in (:local, :global, :export)
1343 print(io, head, ' ')

1371 show_block(io, "finally", Any[], args[4], indent)
1372 end
1373 print(io, "end")
1374

1377 print(io, "end")

1422 parens && print(io, ")")

1427 show_unquoted(io, args[1])
1428 else

1430 show_unquoted(io, args[1])
1431 print(io, ")")
1432 end

1471 unhandled = true
1472 end

1874 """
1875 showarg(io::IO, x, toplevel)
1876
1877 Show `x` as if it were an argument to a function. This function is
1878 used by [`summary`](@ref) to display type information in terms of sequences of
1879 function calls on objects. `toplevel` is `true` if this is
1880 the direct call from `summary` and `false` for nested (recursive) calls.
1881
1882 The fallback definition is to print `x` as "::\\\$(typeof(x))",
1883 representing argument `x` in terms of its type. (The double-colon is
1884 omitted if `toplevel=true`.) However, you can
1885 specialize this function for specific types to customize printing.
1886
1887 # Example
1888
1889 A SubArray created as `view(a, :, 3, 2:5)`, where `a` is a
1890 3-dimensional Float64 array, has type
1891
1892 SubArray{Float64,2,Array{Float64,3},Tuple{Colon,Int64,UnitRange{Int64}},false}
1893
1894 The default `show` printing would display this full type.
1895 However, the summary for SubArrays actually prints as
1896
1897 2×4 view(::Array{Float64,3}, :, 3, 2:5) with eltype Float64
1898
1899 because of a definition similar to
1900
1901 function Base.showarg(io::IO, v::SubArray, toplevel)
1902 print(io, "view(")
1903 showarg(io, parent(v), false)
1904 print(io, ", ", join(v.indices, ", "))
1905 print(io, ')')
1906 toplevel && print(io, " with eltype ", eltype(v))
1907 end
1908
1909 Note that we're calling `showarg` recursively for the parent array
1910 type, indicating that any recursed calls are not at the top level.
1911 Printing the parent as `::Array{Float64,3}` is the fallback (non-toplevel)
1912 behavior, because no specialized method for `Array` has been defined.
1913 """
