193 """
194 Timer(callback::Function, delay; interval = 0)
195
196 Create a timer that wakes up tasks waiting for it (by calling [`wait`](@ref) on the timer object) and
197 calls the function `callback`.
198
199 Waiting tasks are woken and the function `callback` is called after an initial delay of `delay` seconds,
200 and then repeating with the given `interval` in seconds. If `interval` is equal to `0`, the timer
201 is only triggered once. The function `callback` is called with a single argument, the timer itself.
202 When the timer is closed (by [`close`](@ref) waiting tasks are woken with an error. Use [`isopen`](@ref)
203 to check whether a timer is still active.
204
205 # Examples
206
207 Here the first number is printed after a delay of two seconds, then the following numbers are printed quickly.
208
209 ```julia-repl
210 julia> begin
211 i = 0
212 cb(timer) = (global i += 1; println(i))
213 t = Timer(cb, 2, interval = 0.2)
214 wait(t)
215 sleep(0.5)
216 close(t)
217 end
218 1
219 2
220 3
221 ```
222 """

224 t = Timer(timeout, interval = interval)

227 success = try
228 wait(t)

235 end

244 timedwait(testcb::Function, secs::Float64; pollint::Float64=0.1)
245
246 Waits until `testcb` returns `true` or for `secs` seconds, whichever is earlier.
247 `testcb` is polled every `pollint` seconds.
248 """
249 function timedwait(testcb::Function, secs::Float64; pollint::Float64=0.1)
250 pollint > 0 || throw(ArgumentError("cannot set pollint to $pollint seconds"))
251 start = time()
252 done = Channel(1)
253 timercb(aw) = begin
254 try
255 if testcb()
256 put!(done, :ok)
257 elseif (time() - start) > secs
258 put!(done, :timed_out)
259 end
260 catch e
261 put!(done, :error)
262 finally
263 isready(done) && close(aw)
264 end
265 end
266
267 if !testcb()
268 t = Timer(timercb, pollint, interval = pollint)
269 ret = fetch(done)
270 close(t)
271 else
272 ret = :ok
273 end
