228 # Convert seconds since epoch to MS-DOS time/date, which has
229 # a resolution of 2 seconds.
230 function _msdostime(secs::Float64)
231 t = Libc.TmStruct(secs)
232 dostime = UInt16((t.hour<<11) | (t.min<<5) | div(t.sec, 2))
233 dosdate = UInt16(((t.year+1900-1980)<<9) | ((t.month+1)<<5) | t.mday)
234 dostime, dosdate
235 end

237 # Convert MS-DOS time/date to seconds since epoch
238 function _mtime(dostime::UInt16, dosdate::UInt16)
239 sec = 2*(dostime & 0x1f)
240 min = (dostime>>5) & 0x3f
241 hour = dostime>>11
242 mday = dosdate & 0x1f
243 month = ((dosdate>>5) & 0xf) - 1
244 year = (dosdate>>9) + 1980 - 1900
245 time(Libc.TmStruct(sec, min, hour, mday, month, year, 0, 0, -1))
246 end

253 "Load a little endian `UInt32` from a `UInt8` vector `b` starting from index `i`"
254 function getindex_u32le(b::Vector{UInt8}, i)
255 b0 = UInt32(b[i])
256 b1 = UInt32(b[i + 1])
257 b2 = UInt32(b[i + 2])
258 b3 = UInt32(b[i + 3])
259 return (b3 << 24) | (b2 << 16) | (b1 << 8) | b0
