269 # seen so far.

272 # If A is grouped, so that each unique element is in a contiguous group, then we only
273 # need to keep track of one element at a time. We replace the elements of A with the
274 # unique elements that we see in the order that we see them. Once we have iterated
275 # through A, we resize A based on the number of unique elements that we see.
276 function _groupedunique!(A::AbstractVector)
277 isempty(A) && return A
278 idxs = eachindex(A)
279 y = first(A)
280 # We always keep the first element
281 it = iterate(idxs, iterate(idxs)[2])
282 count = 1
283 for x in Iterators.drop(A, 1)
284 if !isequal(x, y)
285 y = A[it[1]] = x
286 count += 1
287 it = iterate(idxs, it[2])
288 end
289 end
290 resize!(A, count)
291 end

587 function _replace!(new::Callable, res::T, A::T,
588 count::Int) where T<:Union{AbstractDict,AbstractSet}
589 c = 0
590 if res === A # cannot replace elements while iterating over A
591 repl = Pair{eltype(A),eltype(A)}[]
592 for x in A
593 y = new(x)
594 if x !== y
595 push!(repl, x => y)
596 c += 1
597 end
598 c == count && break
599 end
600 for oldnew in repl
601 pop!(res, askey(first(oldnew), res))
602 end
603 for oldnew in repl
604 push!(res, last(oldnew))
605 end
606 else
607 for x in A
608 y = new(x)
609 if x !== y
610 pop!(res, askey(x, res))
611 push!(res, y)
612 c += 1
613 end
614 c == count && break
615 end
616 end
617 res
