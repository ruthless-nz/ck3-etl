59 # convert dual arrays (ips, interpreter_frames) to a single array of locations
60 function _reformat_bt(bt, bt2)
61 ret = Vector{Union{InterpreterIP,Ptr{Cvoid}}}()
62 i, j = 1, 1
63 while i <= length(bt)
64 ip = bt[i]::Ptr{Cvoid}
65 if ip == Ptr{Cvoid}(-1%UInt)
66 # The next one is really a CodeInfo
67 push!(ret, InterpreterIP(
68 bt2[j],
69 bt[i+2]))
70 j += 1
71 i += 3
72 else
73 push!(ret, Ptr{Cvoid}(ip))
74 i += 1
75 end
76 end
77 ret
78 end

94 """
95 catch_stack(task=current_task(); [inclue_bt=true])
96
97 Get the stack of exceptions currently being handled. For nested catch blocks
98 there may be more than one current exception in which case the most recently
99 thrown exception is last in the stack. The stack is returned as a Vector of
100 `(exception,backtrace)` pairs, or a Vector of exceptions if `include_bt` is
101 false.
102
103 Explicitly passing `task` will return the current exception stack on an
104 arbitrary task. This is useful for inspecting tasks which have failed due to
105 uncaught exceptions.
106
107 !!! compat "Julia 1.1"
108 This function is experimental in Julia 1.1 and will likely be renamed in a
109 future release (see https://github.com/JuliaLang/julia/pull/29901).
110 """
111 function catch_stack(task=current_task(); include_bt=true)
112 raw = ccall(:jl_get_excstack, Any, (Any,Cint,Cint), task, include_bt, typemax(Cint))
113 formatted = Any[]
114 stride = include_bt ? 3 : 1
115 for i = reverse(1:stride:length(raw))
116 e = raw[i]
117 push!(formatted, include_bt ? (e,Base._reformat_bt(raw[i+1],raw[i+2])) : e)
118 end
119 formatted
120 end

154 """
155 @assert cond [text]
156
157 Throw an [`AssertionError`](@ref) if `cond` is `false`. Preferred syntax for writing assertions.
158 Message `text` is optionally displayed upon assertion failure.
159
160 !!! warning
161 An assert might be disabled at various optimization levels.
162 Assert should therefore only be used as a debugging tool
163 and not used for authentication verification (e.g., verifying passwords),
164 nor should side effects needed for the function to work correctly
165 be used inside of asserts.
166
167 # Examples
168 ```jldoctest
169 julia> @assert iseven(3) "3 is an odd number!"
170 ERROR: AssertionError: 3 is an odd number!
171
172 julia> @assert isodd(3) "What even are numbers?"
173 ```
174 """
175 macro assert(ex, msgs...)
176 msg = isempty(msgs) ? ex : msgs[1]
177 if isa(msg, AbstractString)
178 msg = msg # pass-through
179 elseif !isempty(msgs) && (isa(msg, Expr) || isa(msg, Symbol))
180 # message is an expression needing evaluating
181 msg = :(Main.Base.string($(esc(msg))))
182 elseif isdefined(Main, :Base) && isdefined(Main.Base, :string) && applicable(Main.Base.string, msg)
183 msg = Main.Base.string(msg)
184 else
185 # string() might not be defined during bootstrap
186 msg = :(Main.Base.string($(Expr(:quote,msg))))
187 end
188 return :($(esc(ex)) ? $(nothing) : throw(AssertionError($msg)))
189 end

204 """
205 ExponentialBackOff(; n=1, first_delay=0.05, max_delay=10.0, factor=5.0, jitter=0.1)
206
207 A [`Float64`](@ref) iterator of length `n` whose elements exponentially increase at a
208 rate in the interval `factor` * (1 ± `jitter`). The first element is
209 `first_delay` and all elements are clamped to `max_delay`.

212 ExponentialBackOff(n, first_delay, max_delay, factor, jitter)
213 function iterate(ebo::ExponentialBackOff, state= (ebo.n, min(ebo.first_delay, ebo.max_delay)))
214 state[1] < 1 && return nothing
215 next_n = state[1]-1
216 curr_delay = state[2]
217 next_delay = min(ebo.max_delay, state[2] * ebo.factor * (1.0 - ebo.jitter + (rand(Float64) * 2.0 * ebo.jitter)))
218 (curr_delay, (next_n, next_delay))
219 end
220 length(ebo::ExponentialBackOff) = ebo.n
221 eltype(::Type{ExponentialBackOff}) = Float64

223 """
224 retry(f; delays=ExponentialBackOff(), check=nothing) -> Function
225
226 Return an anonymous function that calls function `f`. If an exception arises,
227 `f` is repeatedly called again, each time `check` returns `true`, after waiting the
228 number of seconds specified in `delays`. `check` should input `delays`'s
229 current state and the `Exception`.
230
231 !!! compat "Julia 1.2"
232 Before Julia 1.2 this signature was restricted to `f::Function`.
233
234 # Examples
235 ```julia
236 retry(f, delays=fill(5.0, 3))
237 retry(f, delays=rand(5:10, 2))
238 retry(f, delays=Base.ExponentialBackOff(n=3, first_delay=5, max_delay=1000))
239 retry(http_get, check=(s,e)->e.status == "503")(url)
240 retry(read, check=(s,e)->isa(e, IOError))(io, 128; all=false)
241 ```
242 """
243 function retry(f; delays=ExponentialBackOff(), check=nothing)
244 (args...; kwargs...) -> begin
245 y = iterate(delays)
246 while y !== nothing
247 (delay, state) = y
248 try
249 return f(args...; kwargs...)
250 catch e
251 y === nothing && rethrow()
252 if check !== nothing
253 result = check(state, e)
254 state, retry_or_not = length(result) == 2 ? result : (state, result)
255 retry_or_not || rethrow()
256 end
257 end
258 sleep(delay)
259 y = iterate(delays, state)
260 end
261 # When delays is out, just run the function without try/catch
262 return f(args...; kwargs...)
263 end
264 end
