71 # Filesystem.open, not Base.open
72 function open(path::AbstractString, flags::Integer, mode::Integer=0)
73 req = Libc.malloc(_sizeof_uv_fs)
74 local handle
75 try
76 ret = ccall(:jl_uv_fs_open, Int32,
77 (Ptr{Cvoid}, Ptr{Cvoid}, Cstring, Int32, Int32, Ptr{Cvoid}),
78 eventloop(), req, path, flags, mode, C_NULL)
79 handle = ccall(:jl_uv_fs_result, Cssize_t, (Ptr{Cvoid},), req)
80 ccall(:uv_fs_req_cleanup, Cvoid, (Ptr{Cvoid},), req)
81 uv_error("open", ret)
82 finally # conversion to Cstring could cause an exception
83 Libc.free(req)
84 end
85 return File(OS_HANDLE(@static Sys.iswindows() ? Ptr{Cvoid}(handle) : Cint(handle)))
86 end

96 function close(f::File)
97 check_open(f)
98 err = ccall(:jl_fs_close, Int32, (OS_HANDLE,), f.handle)
99 uv_error("close", err)
100 f.handle = INVALID_OS_HANDLE
101 f.open = false
102 return nothing

160 function read(f::File, ::Type{Char})
161 b0 = read(f, UInt8)
162 l = 8(4-leading_ones(b0))
163 c = UInt32(b0) << 24
164 if l < 24
165 s = 16
166 while s ≥ l && !eof(f)
167 p = position(f)
168 b = read(f, UInt8)
169 if b & 0xc0 != 0x80
170 seek(f, p)
171 break
172 end
173 c |= UInt32(b) << s
174 s -= 8
175 end
176 end
177 return reinterpret(Char, c)
178 end

195 nr = min(nb, bytesavailable(f))
