646 """
647 walkdir(dir; topdown=true, follow_symlinks=false, onerror=throw)
648
649 Return an iterator that walks the directory tree of a directory.
650 The iterator returns a tuple containing `(rootpath, dirs, files)`.
651 The directory tree can be traversed top-down or bottom-up.
652 If `walkdir` encounters a [`SystemError`](@ref)
653 it will rethrow the error by default.
654 A custom error handling function can be provided through `onerror` keyword argument.
655 `onerror` is called with a `SystemError` as argument.
656
657 # Examples
658 ```julia
659 for (root, dirs, files) in walkdir(".")
660 println("Directories in \$root")
661 for dir in dirs
662 println(joinpath(root, dir)) # path to directories
663 end
664 println("Files in \$root")
665 for file in files
666 println(joinpath(root, file)) # path to files
667 end
668 end
669 ```
670
671 ```julia-repl
672 julia> mkpath("my/test/dir");
673
674 julia> itr = walkdir("my");
675
676 julia> (root, dirs, files) = first(itr)
677 ("my", ["test"], String[])
678
679 julia> (root, dirs, files) = first(itr)
680 ("my/test", ["dir"], String[])
681
682 julia> (root, dirs, files) = first(itr)
683 ("my/test/dir", String[], String[])
684 ```
685 """
686 function walkdir(root; topdown=true, follow_symlinks=false, onerror=throw)
687 content = nothing
688 try
689 content = readdir(root)
690 catch err
691 isa(err, SystemError) || throw(err)
692 onerror(err)
693 # Need to return an empty closed channel to skip the current root folder
694 chnl = Channel(0)
695 close(chnl)
696 return chnl
697 end
698 dirs = Vector{eltype(content)}()
699 files = Vector{eltype(content)}()
700 for name in content
701 if isdir(joinpath(root, name))
702 push!(dirs, name)
703 else
704 push!(files, name)
705 end
706 end
707
708 function _it(chnl)
709 if topdown
710 put!(chnl, (root, dirs, files))
711 end
712 for dir in dirs
713 path = joinpath(root,dir)
714 if follow_symlinks || !islink(path)
715 for (root_l, dirs_l, files_l) in walkdir(path, topdown=topdown, follow_symlinks=follow_symlinks, onerror=onerror)
716 put!(chnl, (root_l, dirs_l, files_l))
717 end
718 end
719 end
720 if !topdown
721 put!(chnl, (root, dirs, files))
722 end
723 end
724
725 return Channel(_it)
726 end
