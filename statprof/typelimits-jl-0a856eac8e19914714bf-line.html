30 # try to find `type` somewhere in `comparison` type
31 # at a minimum nesting depth of `mindepth`
32 function is_derived_type(@nospecialize(t), @nospecialize(c), mindepth::Int)
33 if t === c
34 return mindepth <= 1
35 end
36 if isa(c, Union)
37 # see if it is one of the elements of the union
38 return is_derived_type(t, c.a, mindepth) || is_derived_type(t, c.b, mindepth)
39 elseif isa(c, UnionAll)
40 # see if it is derived from the body
41 # also handle the var here, since this construct bounds the mindepth to the smallest possible value
42 return is_derived_type(t, c.var.ub, mindepth) || is_derived_type(t, c.body, mindepth)
43 elseif isa(c, DataType)
44 if mindepth > 0
45 mindepth -= 1
46 end
47 if isa(t, DataType)
48 # see if it is one of the supertypes of a parameter
49 super = supertype(c)
50 while super !== Any
51 t === super && return true
52 super = supertype(super)
53 end
54 end
55 # see if it was extracted from a type parameter
56 cP = c.parameters
57 for p in cP
58 is_derived_type(t, p, mindepth) && return true
59 end
60 if isconcretetype(c) && isbitstype(c)
61 # see if it was extracted from a fieldtype
62 # however, only look through types that can be inlined
63 # to ensure monotonicity of derivation
64 # since we know that for immutable, concrete, bits types,
65 # the field types must have been constructed prior to the type,
66 # it cannot have a reference cycle in the type graph
67 cF = c.types
68 for f in cF
69 # often a parameter is also a field type; avoid searching twice
70 if !contains_is(c.parameters, f)
71 is_derived_type(t, f, mindepth) && return true
72 end
73 end
74 end
75 end
76 return false
77 end

86 # The goal of this function is to return a type of greater "size" and less "complexity" than
87 # both `t` or `c` over the lattice defined by `sources`, `depth`, and `allowed_tuplelen`.
88 function _limit_type_size(@nospecialize(t), @nospecialize(c), sources::SimpleVector, depth::Int, allowed_tuplelen::Int)

92 return t # easy case

95 else

97 if isa(ut, DataType) && ut.name !== _va_typename && isa(c, Type) && c !== Union{} && c <: t
98 return t # t is already wider than the comparison in the type lattice

100 return t # t isn't something new
101 end
102 end
103 # peel off (and ignore) wrappers - they contribute no useful information, so we don't need to consider their size
104 # first attempt to turn `c` into a type that contributes meaningful information
105 # by peeling off meaningless non-matching wrappers of comparison one at a time
106 # then unwrap `t`
107 if isa(c, TypeVar)
108 if isa(t, TypeVar) && t.ub === c.ub && (t.lb === Union{} || t.lb === c.lb)

111 return _limit_type_size(t, c.ub, sources, depth, allowed_tuplelen)
112 end
113 if isa(c, UnionAll)

115 end
116 if isa(t, UnionAll)

118 tbody === t.body && return t

120 elseif isa(t, TypeVar)
121 # don't have a matching TypeVar in comparison, so we keep just the upper bound
122 return _limit_type_size(t.ub, c, sources, depth, allowed_tuplelen)
123 elseif isa(t, Union)
124 if isa(c, Union)
125 a = _limit_type_size(t.a, c.a, sources, depth, allowed_tuplelen)
126 b = _limit_type_size(t.b, c.b, sources, depth, allowed_tuplelen)
127 return Union{a, b}
128 end
129 elseif isa(t, DataType)
130 if isa(c, DataType)
131 tP = t.parameters
132 cP = c.parameters
133 if t.name === c.name && !isempty(cP)
134 if isvarargtype(t)

136 N = tP[2]

141 elseif t.name === Tuple.name
142 # for covariant datatypes (Tuple),
143 # apply type-size limit element-wise
144 ltP = length(tP)
145 lcP = length(cP)

147 Q = Any[ tP[i] for i in 1:np ]
148 if ltP > np
149 # combine tp[np:end] into tP[np] using Vararg
150 Q[np] = tuple_tail_elem(Bottom, Any[ tP[i] for i in np:ltP ])
151 end
152 for i = 1:np
153 # now apply limit element-wise to Q
154 # padding out the comparison as needed to allowed_tuplelen elements
155 if i <= lcP
156 cPi = cP[i]
157 elseif isvarargtype(cP[lcP])
158 cPi = cP[lcP]
159 else
160 cPi = Any
161 end
162 Q[i] = _limit_type_size(Q[i], cPi, sources, depth + 1, 0)
163 end
164 return Tuple{Q...}
165 end
166 elseif isvarargtype(c)
167 # Tuple{Vararg{T}} --> Tuple{T} is OK
168 return _limit_type_size(t, cP[1], sources, depth, 0)
169 end
170 end
171 if isType(t) # allow taking typeof as Type{...}, but ensure it doesn't start nesting
172 tt = unwrap_unionall(t.parameters[1])
173 if isa(tt, DataType) && !isType(tt)
174 is_derived_type_from_any(tt, sources, depth) && return t
175 end
176 end
177 if isvarargtype(t)
178 # never replace Vararg with non-Vararg
179 return Vararg
180 end
181 if allowed_tuplelen < 1 && t.name === Tuple.name
182 return Any
183 end
184 widert = t.name.wrapper
185 if !(t <: widert)
186 # This can happen when a typevar has bounds too wide for its context, e.g.
187 # `Complex{T} where T` is not a subtype of `Complex`. In that case widen even
188 # faster to something safe to ensure the result is a supertype of the input.

270 elseif isvarargtype(c)
271 return type_more_complex(t, unwrapva(c), sources, depth, tupledepth, 0)
272 end
273 if isType(t) # allow taking typeof any source type anywhere as Type{...}, as long as it isn't nesting Type{Type{...}}
274 tt = unwrap_unionall(t.parameters[1])
275 if isa(tt, DataType) && !isType(tt)
