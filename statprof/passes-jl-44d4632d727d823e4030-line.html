3 """
4 This struct keeps track of all uses of some mutable struct allocated
5 in the current function. `uses` are all instances of `getfield` on the
6 struct. `defs` are all instances of `setfield!` on the struct. The terminology
7 refers to the uses/defs of the ``slot bundle'' that the mutable struct represents.
8
9 In addition we keep track of all instances of a foreigncall preserve of this mutable
10 struct. Somewhat counterintuitively, we don't actually need to make sure that the
11 struct itself is live (or even allocated) at a ccall site. If there are no other places
12 where the struct escapes (and thus e.g. where its address is taken), it need not be
13 allocated. We do however, need to make sure to preserve any elements of this struct.
14 """

29 function lift_defuse(cfg::CFG, ssa::SSADefUse)
30 # We remove from `uses` any block where all uses are dominated
31 # by a def. This prevents insertion of dead phi nodes at the top
32 # of such a block if that block happens to be in a loop
33 ordered = Tuple{Int, Int, Bool}[(x, block_for_inst(cfg, x), true) for x in ssa.uses]
34 for x in ssa.defs
35 push!(ordered, (x, block_for_inst(cfg, x), false))
36 end
37 ordered = sort(ordered, by=x->x[1])
38 bb_defs = Int[]
39 bb_uses = Int[]
40 last_bb = last_def_bb = 0
41 for (_, bb, is_use) in ordered
42 if bb != last_bb && is_use
43 push!(bb_uses, bb)
44 end
45 last_bb = bb
46 if last_def_bb != bb && !is_use
47 push!(bb_defs, bb)
48 last_def_bb = bb
49 end
50 end
51 SSADefUse(bb_uses, bb_defs, Int[])
52 end

1134 # Do nothing
1135 else
1136 process_node!(compact, compact.result_idx, stmt, i, i, ms, true)
1137 end
1138 # We always increase the result index to ensure a predicatable
1139 # placement of the resulting nodes.
1140 compact.result_idx += 1
1141 end
1142 ms = merged_succ[ms]
1143 end
1144 end
1145
1146 compact.active_result_bb = length(bb_starts)
