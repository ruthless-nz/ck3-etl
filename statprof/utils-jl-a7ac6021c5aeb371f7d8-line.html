228 # Return a 3-tuple of a permutation that sorts rows into groups,
229 # and the positions of the first and last rows in each group in that permutation
230 # `groups` must contain group indices in 0:ngroups
231 # Rows with group index 0 are skipped (used when skipmissing=true)
232 # Partly uses the code of Wes McKinney's groupsort_indexer in pandas (file: src/groupby.pyx).
233 function compute_indices(groups::AbstractVector{<:Integer}, ngroups::Integer)
234 # count elements in each group
235 stops = zeros(Int, ngroups+1)
236 @inbounds for gix in groups
237 stops[gix+1] += 1

246 end
247 end
