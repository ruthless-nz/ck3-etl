949 """
950 may_invoke_generator(method, atypes, sparams)
951
952 Computes whether or not we may invoke the generator for the given `method` on
953 the given atypes and sparams. For correctness, all generated function are
954 required to return monotonic answers. However, since we don't expect users to
955 be able to successfully implement this criterion, we only call generated
956 functions on concrete types. The one exception to this is that we allow calling
957 generators with abstract types if the generator does not use said abstract type
958 (and thus cannot incorrectly use it to break monotonicity). This function
959 computes whether we are in either of these cases.
960
961 Unlike normal functions, the compilation heuristics still can't generate good dispatch
962 in some cases, but this may still allow inference not to fall over in some limited cases.
963 """
964 function may_invoke_generator(method::MethodInstance)

966 end
967 function may_invoke_generator(method::Method, @nospecialize(atypes), sparams::SimpleVector)
968 # If we have complete information, we may always call the generator
969 isdispatchtuple(atypes) && return true
970
971 # We don't have complete information, but it is possible that the generator
972 # syntactically doesn't make use of the information we don't have. Check
973 # for that.
974
975
61 (0.27%) samples spent in assemble_inline_todo!
61 (100.00%) (ex.), 61 (100.00%) (incl.) when called from ssa_inlining_pass! line 70
# For now, only handle the (common, generated by the frontend case) that the
976 # generator only has one method
977 isa(method.generator, Core.GeneratedFunctionStub) || return false
978 gen_mthds = methods(method.generator.gen)
979 length(gen_mthds) == 1 || return false
980
981 generator_method = first(gen_mthds)
982 nsparams = length(sparams)
983 isdefined(generator_method, :source) || return false
984 code = generator_method.source
985 nslots = ccall(:jl_ast_nslots, Int, (Any,), code)
986 at = unwrap_unionall(atypes)
987 (nslots >= 1 + length(sparams) + length(at.parameters)) || return false
988
989 for i = 1:nsparams
990 if isa(sparams[i], TypeVar)
991 if (ast_slotflag(code, 1 + i) & SLOT_USED) != 0
992 return false
993 end
994 end
995 end
996 for i = 1:length(at.parameters)
997 if !isdispatchelem(at.parameters[i])
998 if (ast_slotflag(code, 1 + i + nsparams) & SLOT_USED) != 0
999 return false
1000 end
1001 end
1002 end
1003 return true
1004 end
