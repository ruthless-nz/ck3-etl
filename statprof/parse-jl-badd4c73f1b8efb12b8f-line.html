7 """
8 parse(type, str; base)
9
10 Parse a string as a number. For `Integer` types, a base can be specified
11 (the default is 10). For floating-point types, the string is parsed as a decimal
12 floating-point number. `Complex` types are parsed from decimal strings
13 of the form `"R±Iim"` as a `Complex(R,I)` of the requested type; `"i"` or `"j"` can also be
14 used instead of `"im"`, and `"R"` or `"Iim"` are also permitted.
15 If the string does not contain a valid number, an error is raised.
16
17 !!! compat "Julia 1.1"
18 `parse(Bool, str)` requires at least Julia 1.1.
19
20 # Examples
21 ```jldoctest
22 julia> parse(Int, "1234")

24
25 julia> parse(Int, "1234", base = 5)

27
28 julia> parse(Int, "afc", base = 16)

30
31 julia> parse(Float64, "1.2e-3")
32 0.0012
33
34 julia> parse(Complex{Float64}, "3.2e-1 + 4.5im")

36 ```

294 function tryparse_internal(::Type{Complex{T}}, s::Union{String,SubString{String}}, i::Int, e::Int, raise::Bool) where {T<:Real}
295 # skip initial whitespace
296 while i ≤ e && isspace(s[i])
297 i = nextind(s, i)
298 end
299 if i > e
300 raise && throw(ArgumentError("input string is empty or only contains whitespace"))
301 return nothing
302 end
303
304 # find index of ± separating real/imaginary parts (if any)
305 i₊ = something(findnext(in(('+','-')), s, i), 0)
306 if i₊ == i # leading ± sign
307 i₊ = something(findnext(in(('+','-')), s, i₊+1), 0)
308 end
309 if i₊ != 0 && s[i₊-1] in ('e','E') # exponent sign
310 i₊ = something(findnext(in(('+','-')), s, i₊+1), 0)
311 end
312
313 # find trailing im/i/j
314 iᵢ = something(findprev(in(('m','i','j')), s, e), 0)
315 if iᵢ > 0 && s[iᵢ] == 'm' # im
316 iᵢ -= 1
317 if s[iᵢ] != 'i'
318 raise && throw(ArgumentError("expected trailing \"im\", found only \"m\""))
319 return nothing
320 end
321 end
322
323 if i₊ == 0 # purely real or imaginary value
324 if iᵢ > i && !(iᵢ == i+1 && s[i] in ('+','-')) # purely imaginary (not "±inf")
325 x = tryparse_internal(T, s, i, iᵢ-1, raise)
326 x === nothing && return nothing
327 return Complex{T}(zero(x),x)
328 else # purely real
329 x = tryparse_internal(T, s, i, e, raise)
330 x === nothing && return nothing
331 return Complex{T}(x)
332 end
333 end
334
335 if iᵢ < i₊
336 raise && throw(ArgumentError("missing imaginary unit"))
337 return nothing # no imaginary part
338 end
339
340 # parse real part
341 re = tryparse_internal(T, s, i, i₊-1, raise)
342 re === nothing && return nothing
343
344 # parse imaginary part
345 im = tryparse_internal(T, s, i₊+1, iᵢ-1, raise)
346 im === nothing && return nothing
347
348 return Complex{T}(re, s[i₊]=='-' ? -im : im)
349 end
