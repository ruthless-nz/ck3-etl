10 """
11 Channel{T}(sz::Int)
12
13 Constructs a `Channel` with an internal buffer that can hold a maximum of `sz` objects
14 of type `T`.
15 [`put!`](@ref) calls on a full channel block until an object is removed with [`take!`](@ref).
16
17 `Channel(0)` constructs an unbuffered channel. `put!` blocks until a matching `take!` is called.
18 And vice-versa.
19
20 Other constructors:
21
22 * `Channel(Inf)`: equivalent to `Channel{Any}(typemax(Int))`
23 * `Channel(sz)`: equivalent to `Channel{Any}(sz)`
24 """

35 function Channel{T}(sz::Integer) where T
36 if sz < 0
37 throw(ArgumentError("Channel size must be either 0, a positive integer or Inf"))
38 end
39 lock = ReentrantLock()
40 cond_put, cond_take = Threads.Condition(lock), Threads.Condition(lock)
41 cond_wait = (sz == 0 ? Threads.Condition(lock) : cond_take) # wait is distinct from take iff unbuffered
42 return new(cond_take, cond_wait, cond_put, :open, nothing, Vector{T}(), sz)
43 end

99 function Channel(func::Function; ctype=Any, csize=0, taskref=nothing)
100 chnl = Channel{ctype}(csize)
101 task = Task(() -> func(chnl))
102 bind(chnl, task)
103 yield(task) # immediately start it
104
105 isa(taskref, Ref{Task}) && (taskref[] = task)
106 return chnl
107 end

121 """
122 close(c::Channel[, excp::Exception])
123
124 Close a channel. An exception (optionally given by `excp`), is thrown by:
125
126 * [`put!`](@ref) on a closed channel.
127 * [`take!`](@ref) and [`fetch`](@ref) on an empty, closed channel.
128 """
129 function close(c::Channel, excp::Exception=closed_exception())
130 lock(c)
131 try
132 c.state = :closed
133 c.excp = excp
134 notify_error(c.cond_take, excp)
135 notify_error(c.cond_wait, excp)
136 notify_error(c.cond_put, excp)
137 finally
138 unlock(c)
139 end

144 """
145 bind(chnl::Channel, task::Task)
146
147 Associate the lifetime of `chnl` with a task.
148 `Channel` `chnl` is automatically closed when the task terminates.
149 Any uncaught exception in the task is propagated to all waiters on `chnl`.
150
151 The `chnl` object can be explicitly closed independent of task termination.
152 Terminating tasks have no effect on already closed `Channel` objects.
153
154 When a channel is bound to multiple tasks, the first task to terminate will
155 close the channel. When multiple channels are bound to the same task,
156 termination of the task will close all of the bound channels.
157
158 # Examples
159 ```jldoctest
160 julia> c = Channel(0);
161
162 julia> task = @async foreach(i->put!(c, i), 1:4);
163
164 julia> bind(c,task);
165
166 julia> for i in c
167 @show i
168 end;
169 i = 1
170 i = 2
171 i = 3
172 i = 4
173
174 julia> isopen(c)
175 false
176 ```
177
178 ```jldoctest
179 julia> c = Channel(0);
180
181 julia> task = @async (put!(c,1);error("foo"));
182
183 julia> bind(c,task);
184
185 julia> take!(c)
186 1
187
188 julia> put!(c,1);
189 ERROR: foo
190 Stacktrace:
191 [...]
192 ```

202 """
203 channeled_tasks(n::Int, funcs...; ctypes=fill(Any,n), csizes=fill(0,n))
204
205 A convenience method to create `n` channels and bind them to tasks started
206 from the provided functions in a single call. Each `func` must accept `n` arguments
207 which are the created channels. Channel types and sizes may be specified via
208 keyword arguments `ctypes` and `csizes` respectively. If unspecified, all channels are
209 of type `Channel{Any}(0)`.
210
211 Returns a tuple, `(Array{Channel}, Array{Task})`, of the created channels and tasks.
212 """
213 function channeled_tasks(n::Int, funcs...; ctypes=fill(Any,n), csizes=fill(0,n))
214 @assert length(csizes) == n
215 @assert length(ctypes) == n
216
217 chnls = map(i -> Channel{ctypes[i]}(csizes[i]), 1:n)
218 tasks = Task[ Task(() -> f(chnls...)) for f in funcs ]
219
220 # bind all tasks to all channels and schedule them
221 foreach(t -> foreach(c -> bind(c, t), chnls), tasks)
222 foreach(schedule, tasks)
223 yield() # Allow scheduled tasks to run
224
225 return (chnls, tasks)
226 end

278 function put_buffered(c::Channel, v)
279 lock(c)

294 function put_unbuffered(c::Channel, v)
295 lock(c)
296 taker = try
297 while isempty(c.cond_take.waitq)
298 check_channel_state(c)
299 notify(c.cond_wait)
300 wait(c.cond_put)
301 end
302 # unfair scheduled version of: notify(c.cond_take, v, false, false); yield()
303 popfirst!(c.cond_take.waitq)
304 finally
305 unlock(c)
306 end
307 # unfair version of: schedule(taker, v); yield()
308 yield(taker, v) # immediately give taker a chance to run, but don't block the current task
309 return v
310 end
