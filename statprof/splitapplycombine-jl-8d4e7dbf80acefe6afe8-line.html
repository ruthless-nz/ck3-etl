5 into row groups.
6
7 # Arguments
8 - `df` : an `AbstractDataFrame` to split
9 - `cols` : data frame columns to group by. Can be any column selector
10 ($COLUMNINDEX_STR; $MULTICOLUMNINDEX_STR).
11 - `sort` : whether to sort groups according to the values of the grouping columns
12 `cols`; if all `cols` are `CategoricalVector`s then groups are always sorted
13 irrespective of the value of `sort`
14 - `skipmissing` : whether to skip groups with `missing` values in one of the
15 grouping columns `cols`
16
17 # Details
18 An iterator over a `GroupedDataFrame` returns a `SubDataFrame` view
19 for each grouping into `df`.
20 Within each group, the order of rows in `df` is preserved.
21
22 `cols` can be any valid data frame indexing expression.
23 In particular if it is an empty vector then a single-group `GroupedDataFrame`
24 is created.
25
26 A `GroupedDataFrame` also supports
27 indexing by groups, `map` (which applies a function to each group)
28 and `combine` (which applies a function to each group
29 and combines the result into a data frame).
30
31 `GroupedDataFrame` also supports the dictionary interface. The keys are
32 [`GroupKey`](@ref) objects returned by [`keys(::GroupedDataFrame)`](@ref),
33 which can also be used to get the values of the grouping columns for each group.
34 `Tuples` and `NamedTuple`s containing the values of the grouping columns (in the
35 same order as the `cols` argument) are also accepted as indices, but this will
36 be slower than using the equivalent `GroupKey`.

597 "is not equal to grouping key :$key"))
598 end
599 end
600 end
601 if keeprows
602 newparent = select(parent(gd), gd.cols, copycols=copycols)
603 else
604 newparent = parent(gd)[idx, gd.cols]
605 end
606 hcat!(newparent,
607 select(valscat, Not(intersect(keys, _names(valscat))), copycols=false),
608 copycols=false)
609 ungroup && return newparent
610
611 if length(idx) == 0 && !(keeprows && length(keys) > 0)
612 @assert nrow(newparent) == 0
613 return GroupedDataFrame(newparent, collect(1:length(gd.cols)), Int[],
614 Int[], Int[], Int[], 0, Dict{Any,Int}(),
615 Threads.ReentrantLock())
616 elseif keeprows
617 @assert length(keys) > 0 || idx == gd.idx
618 @assert names(newparent, 1:length(gd.cols)) == names(parent(gd), gd.cols)
619 # in this case we are sure that the result GroupedDataFrame has the
620 # same structure as the source except that grouping columns are at the start
621 Threads.lock(gd.lazy_lock)
622 new_gd = GroupedDataFrame(newparent, collect(1:length(gd.cols)), gd.groups,
623 getfield(gd, :idx), getfield(gd, :starts),
624 getfield(gd, :ends), gd.ngroups,
625 getfield(gd, :keymap), Threads.ReentrantLock())
626 Threads.lock(gd.lazy_lock)
627 return new_gd
628 else
629 groups = gen_groups(idx)
630 @assert groups[end] <= length(gd)
631 @assert names(newparent, 1:length(gd.cols)) == names(parent(gd), gd.cols)
632 return GroupedDataFrame(newparent, collect(1:length(gd.cols)), groups,
633 nothing, nothing, nothing, groups[end], nothing,
634 Threads.ReentrantLock())
635 end
636 else
637 if keeprows
638 if nrow(parent(gd)) > 0
639 throw(ArgumentError("select and transform do not support " *
640 "`GroupedDataFrame`s from which some groups have "*
641 "been dropped (including skipmissing=true)"))
642 end
643 if ungroup
644 return keepkeys ? select(parent(gd), gd.cols, copycols=copycols) : DataFrame()
645 else
646 return GroupedDataFrame(select(parent(gd), gd.cols, copycols=copycols),
647 collect(1:length(gd.cols)),
648 Int[], Int[], Int[], Int[], 0, Dict{Any,Int}(),
649 Threads.ReentrantLock())
650 end
651 else
652 if ungroup
653 return keepkeys ? parent(gd)[1:0, gd.cols] : DataFrame()
654 else
655 return GroupedDataFrame(parent(gd)[1:0, gd.cols], collect(1:length(gd.cols)),
656 Int[], Int[], Int[], Int[], 0, Dict{Any,Int}(),
657 Threads.ReentrantLock())
658 end
659 end
660 end

831 # Find first value matching condition for each group
832 # Optimized for situations where a matching value is typically encountered
833 # among the first rows for each group
834 function fillfirst!(condf, outcol::AbstractVector, incol::AbstractVector,
835 gd::GroupedDataFrame; rev::Bool=false)
836 ngroups = gd.ngroups
837 # Use group indices if they have already been computed
838 idx = getfield(gd, :idx)
839 if idx !== nothing && condf === nothing
840 v = rev ? gd.ends : gd.starts
841 @inbounds for i in 1:ngroups
842 outcol[i] = incol[idx[v[i]]]
843 end
844 elseif idx !== nothing
845 nfilled = 0
846 starts = gd.starts
847 @inbounds for i in eachindex(outcol)
848 s = starts[i]
849 offsets = rev ? (nrow(gd[i])-1:-1:0) : (0:nrow(gd[i])-1)
850 for j in offsets
851 x = incol[idx[s+j]]
852 if !condf === nothing || condf(x)
853 outcol[i] = x
854 nfilled += 1
855 break
856 end
857 end
858 end
859 if nfilled < length(outcol)
860 throw(ArgumentError("some groups contain only missing values"))
861 end
862 else # Finding first row is faster than computing all group indices
863 groups = gd.groups
864 if rev
865 r = length(groups):-1:1
866 else
867 r = 1:length(groups)
868 end
869 filled = fill(false, ngroups)
870 nfilled = 0
871 @inbounds for i in r
872 gix = groups[i]
873 x = incol[i]
874 if gix > 0 && (condf === nothing || condf(x)) && !filled[gix]
875 filled[gix] = true
876 outcol[gix] = x
877 nfilled += 1
878 nfilled == ngroups && break
879 end
880 end
881 if nfilled < length(outcol)
882 throw(ArgumentError("some groups contain only missing values"))
883 end
884 end
885 outcol
886 end

888 # Use a strategy similar to reducedim_init from Base to get the vector of the right type
889 function groupreduce_init(op, condf, adjust,
890 incol::AbstractVector{U}, gd::GroupedDataFrame) where U
891 T = Base.promote_union(U)
892
893 if op === Base.add_sum
894 initf = zero
895 elseif op === Base.mul_prod
896 initf = one
897 else
898 throw(ErrorException("Unrecognized op $op"))
899 end
900
901 Tnm = nonmissingtype(T)
902 if isconcretetype(Tnm) && applicable(initf, Tnm)
903 tmpv = initf(Tnm)
904 initv = op(tmpv, tmpv)
905 x = adjust isa Nothing ? initv : adjust(initv, 1)
906 if condf === !ismissing
907 V = typeof(x)
908 else
909 V = U >: Missing ? Union{typeof(x), Missing} : typeof(x)
910 end
911 v = similar(incol, V, length(gd))
912 fill!(v, x)
913 return v
914 else
915 # do not try to determine the narrowest possible type nor starting value
916 # as this is not possible to do correctly in general without processing
917 # groups; it will get fixed later in groupreduce!; later we
918 # will make use of the fact that this vector is filled with #undef
919 # while above the vector is filled with a concrete value
920 return Vector{Any}(undef, length(gd))
921 end
922 end

926 function groupreduce_init(::typeof($op), condf, adjust,
927 incol::AbstractVector{T}, gd::GroupedDataFrame) where T
928 @assert isnothing(adjust)
929 S = nonmissingtype(T)
930 # !ismissing check is purely an optimization to avoid a copy later
931 outcol = similar(incol, condf === !ismissing ? S : T, length(gd))
932 # Comparison is possible only between CatValues from the same pool
933 if incol isa CategoricalVector
934 U = Union{CategoricalArrays.leveltype(outcol),
935 eltype(outcol) >: Missing ? Missing : Union{}}
936 outcol = CategoricalArray{U, 1}(outcol.refs, incol.pool)
937 end
938 # It is safe to use a non-missing init value
939 # since missing will poison the result if present
940 # we assume here that groups are non-empty (current design assures this)
941 if isconcretetype(S) && hasmethod($initf, Tuple{S})
942 fill!(outcol, $initf(S))
943 else
944 fillfirst!(condf, outcol, incol, gd)
945 end
946 return outcol
947 end

965 function groupreduce!(res::AbstractVector, f, op, condf, adjust, checkempty::Bool,
966 incol::AbstractVector, gd::GroupedDataFrame)
967 n = length(gd)
968 if adjust !== nothing || checkempty
969 counts = zeros(Int, n)
970 end
971 groups = gd.groups
972 @inbounds for i in eachindex(incol, groups)
973 gix = groups[i]
974 x = incol[i]
975 if gix > 0 && (condf === nothing || condf(x))
976 # this check should be optimized out if U is not Any
977 if eltype(res) === Any && !isassigned(res, gix)
978 res[gix] = f(x, gix)
979 else
980 res[gix] = op(res[gix], f(x, gix))
981 end
982 if adjust !== nothing || checkempty
983 counts[gix] += 1
984 end
985 end
986 end
987 # handle the case of an unitialized reduction
988 if eltype(res) === Any
989 if op === Base.add_sum
990 initf = zero
991 elseif op === Base.mul_prod
992 initf = one
993 else
994 initf = x -> throw(ErrorException("Unrecognized op $op"))
995 end
996 @inbounds for gix in eachindex(res)
997 if !isassigned(res, gix)
998 res[gix] = initf(nonmissingtype(eltype(incol)))
999 end
1000 end
1001 end
1002 if adjust !== nothing
1003 res .= adjust.(res, counts)
1004 end
1005 if checkempty && any(iszero, counts)
1006 throw(ArgumentError("some groups contain only missing values"))
1007 end
1008 # Undo pool sharing done by groupreduce_init
1009 if res isa CategoricalVector && res.pool === incol.pool
1010 V = Union{CategoricalArrays.leveltype(res),
1011 eltype(res) >: Missing ? Missing : Union{}}
1012 res = CategoricalArray{V, 1}(res.refs, copy(res.pool))
1013 end
1014 if isconcretetype(eltype(res))
1015 return res
1016 else
1017 return copyto_widen!(Tables.allocatecolumn(typeof(first(res)), n), res)
1018 end
1019 end

1118 @assert all(x -> first(x) isa Union{Int, AbstractVector{Int}, AsTable}, f)
1119 @assert all(x -> last(x) isa Base.Callable, f)

1121 isempty(f) && return Int[], DataFrame()
1122 if keeprows
1123 if minimum(gd.groups) == 0
1124 throw(ArgumentError("select and transform do not support " *

1126 "been dropped (including skipmissing=true)"))

1128 idx_keeprows = prepare_idx_keeprows(gd.idx, gd.starts, gd.ends, nrow(parent(gd)))
1129 else

1136 idx_agg = Vector{Int}(undef, length(gd))

1145 for (i, p) in enumerate(f)
1146 source_cols, fun = p
1147 if isagg(p)
1148 incol = parentdf[!, source_cols]
1149 agg = check_aggregate(last(p))
1150 outcol = agg(incol, gd)

1152 elseif keeprows && fun === identity && !(source_cols isa AsTable)

1162 source_cols.cols,
1163 copycols=false))

1176 if !(firstres isa AbstractVector) && isnothing(idx_agg)
1177 idx_agg = Vector{Int}(undef, length(gd))
1178 fillfirst!(nothing, idx_agg, 1:length(gd.groups), gd)

1188 Val(firstmulticol),
1189 firstres isa AbstractVector ? nothing : idx_agg)

1192 end
1193 end
1194 # idx_agg === nothing then we have only functions that
1195 # returned multiple rows and idx_loc = 1
1196 idx_loc = findfirst(x -> x[1] !== idx_agg, res)
1197 if !keeprows && isnothing(idx_loc)
1198 @assert !isnothing(idx_agg)
1199 idx = idx_agg
1200 else
1201 idx = keeprows ? idx_keeprows : res[idx_loc][1]
1202 agg2idx_map = nothing
1203 for i in 1:length(res)
1204 if res[i][1] !== idx && res[i][1] != idx
1205 if res[i][1] === idx_agg
1206 # we perform pseudo broadcasting here
1207 # keep -1 as a sentinel for errors
1208 if isnothing(agg2idx_map)
1209 agg2idx_map = _agg2idx_map_helper(idx, idx_agg)
1210 end
1211 res[i] = idx_agg, res[i][2][agg2idx_map]
1212 elseif idx != res[i][1]
1213 if keeprows
1214 throw(ArgumentError("all functions must return vectors with " *
1215 "as many values as rows in each group"))
1216 else
1217 throw(ArgumentError("all functions must return vectors of the same length"))
1218 end
1219 end
1220 end
1221 end
1222 end
1223
1224 # here first field in res[i] is used to keep track how the column was generated
1225 # a correct index is stored in idx variable
1226
1227 for (i, (col_idx, col)) in enumerate(res)
1228 if keeprows && res[i][1] !== idx_keeprows # we need to reorder the column
1229 newcol = similar(col)
1230 # we can probably make it more efficient, but I leave it as an optimization for the future
1231 gd_idx = gd.idx
1232 k = 0
1233 for (s, e) in zip(gd.starts, gd.ends)
1234 for j in s:e
1235 k += 1
1236 newcol[gd_idx[j]] = col[k]
1237 end
1238 end
1239 @assert k == length(gd_idx)
1240 res[i] = (col_idx, newcol)
1241 end
1242 end
1243 outcols = map(x -> x[2], res)
1244 # this check is redundant given we check idx above
1245 # but it is safer to double check and it is cheap
1246 @assert all(x -> length(x) == length(outcols[1]), outcols)
1247 return idx, DataFrame!(collect(AbstractVector, outcols), nms)
1248 end

1259 function _combine(p::Pair, gd::GroupedDataFrame, ::Nothing,
1260 copycols::Bool, keeprows::Bool)
1261 # here p should not be normalized as we allow tabular return value from fun
1262 # map and combine should not dispatch here if p is isagg
1263 @assert copycols && !keeprows
1264 source_cols, (fun, out_col) = normalize_selection(index(parent(gd)), p)
1265 parentdf = parent(gd)
1266 if source_cols isa Int
1267 incols = (parent(gd)[!, source_cols],)
1268 elseif source_cols isa AsTable
1269 incols = Tables.columntable(select(parentdf,
1270 source_cols.cols,
1271 copycols=false))
1272 else
1273 @assert source_cols isa AbstractVector{Int}
1274 incols = ntuple(i -> parent(gd)[!, source_cols[i]], length(source_cols))
1275 end
1276 firstres = do_call(fun, gd.idx, gd.starts, gd.ends, gd, incols, 1)
1277 idx, outcols, nms = _combine_multicol(firstres, fun, gd, incols)
1278 # disallow passing target column name to genuine tables
1279 if firstres isa MULTI_COLS_TYPE
1280 if p isa Pair{<:Any, <:Pair{<:Any, <:SymbolOrString}}
1281 throw(ArgumentError("setting column name for tabular return value is disallowed"))
1282 end
1283 else
1284 # fetch auto generated or passed target column name to nms overwritting
1285 # what _combine_with_first produced
1286 nms = [out_col]
1287 end
1288 valscat = DataFrame(collect(AbstractVector, outcols), nms)
1289 return idx, valscat
1290 end

1353 i::Integer, colstart::Integer,
1354 colnames::NTuple{N, Symbol}) where N
1355 if _ncol(row) != N
1356 throw(ArgumentError("return value must have the same number of columns " *
1357 "for all groups (got $N and $(length(row)))"))
1358 end
1359 @inbounds for j in colstart:length(outcols)
1360 col = outcols[j]
1361 cn = colnames[j]
1362 local val
1363 try
1364 val = row[cn]
1365 catch
1366 throw(ArgumentError("return value must have the same column names " *
1367 "for all groups (got $colnames and $(propertynames(row)))"))
1368 end
1369 S = typeof(val)
1370 T = eltype(col)
1371 if S <: T || promote_type(S, T) <: T
1372 col[i] = val
1373 else
1374 return j
1375 end
1376 end
1377 return nothing
1378 end

1380 function _combine_rows_with_first!(first::Union{NamedTuple, DataFrameRow},
1381 outcols::NTuple{N, AbstractVector},
1382 rowstart::Integer, colstart::Integer,
1383 f::Any, gd::GroupedDataFrame,
1384 incols::Union{Nothing, AbstractVector, Tuple, NamedTuple},
1385 colnames::NTuple{N, Symbol},
1386 firstmulticol::Val) where N
1387 len = length(gd)
1388 gdidx = gd.idx
1389 starts = gd.starts
1390 ends = gd.ends
1391 # Handle first group
1392 j = fill_row!(first, outcols, rowstart, colstart, colnames)

1394 # Handle remaining groups
1395 @inbounds for i in rowstart+1:len

1400 let i = i, j = j, outcols=outcols, row=row # Workaround for julia#15276
1401 newcols = ntuple(length(outcols)) do k
1402 S = typeof(row[k])
1403 T = eltype(outcols[k])

1405 if S <: T || U <: T
1406 outcols[k]
1407 else
1408 copyto!(Tables.allocatecolumn(U, length(outcols[k])),
1409 1, outcols[k], 1, k >= j ? i-1 : i)
1410 end
1411 end
1412 end
1413 return _combine_rows_with_first!(row, newcols, i, j,
1414 f, gd, incols, colnames, firstmulticol)
1415 end

1434 function append_rows!(rows, outcols::NTuple{N, AbstractVector},
1435 colstart::Integer, colnames::NTuple{N, Symbol}) where N
1436 if !isa(rows, Union{AbstractDataFrame, NamedTuple{<:Any, <:Tuple{Vararg{AbstractVector}}}})
1437 throw(ArgumentError(ERROR_ROW_COUNT))
1438 elseif _ncol(rows) != N
1439 throw(ArgumentError("return value must have the same number of columns " *
1440 "for all groups (got $N and $(_ncol(rows)))"))
1441 end
1442 @inbounds for j in colstart:length(outcols)
1443 col = outcols[j]
1444 cn = colnames[j]

1446 try
1447 vals = getproperty(rows, cn)

1449 throw(ArgumentError("return value must have the same column names " *

1461 function _combine_tables_with_first!(first::Union{AbstractDataFrame,
1462 NamedTuple{<:Any, <:Tuple{Vararg{AbstractVector}}}},
1463 outcols::NTuple{N, AbstractVector},
1464 idx::Vector{Int}, rowstart::Integer, colstart::Integer,
1465 f::Any, gd::GroupedDataFrame,
1466 incols::Union{Nothing, AbstractVector, Tuple, NamedTuple},
1467 colnames::NTuple{N, Symbol},
1468 firstmulticol::Val) where N
1469 len = length(gd)
1470 gdidx = gd.idx
1471 starts = gd.starts
1472 ends = gd.ends
1473 # Handle first group
1474
1475 @assert _ncol(first) == N
1476 if !isempty(colnames)
1477 j = append_rows!(first, outcols, colstart, colnames)
1478 @assert j === nothing # eltype is guaranteed to match
1479 append!(idx, Iterators.repeated(gdidx[starts[rowstart]], _nrow(first)))
1480 end
1481 # Handle remaining groups
1482 @inbounds for i in rowstart+1:len
1483 rows = wrap_table(do_call(f, gdidx, starts, ends, gd, incols, i), firstmulticol)
1484 _ncol(rows) == 0 && continue
1485 if isempty(colnames)
1486 newcolnames = tuple(propertynames(rows)...)
1487 if rows isa AbstractDataFrame
1488 eltys = eltype.(eachcol(rows))
1489 else
1490 eltys = map(eltype, rows)
1491 end
1492 initialcols = ntuple(i -> Tables.allocatecolumn(eltys[i], 0), _ncol(rows))
1493 return _combine_tables_with_first!(rows, initialcols, idx, i, 1,
1494 f, gd, incols, newcolnames, firstmulticol)
1495 end
1496 j = append_rows!(rows, outcols, 1, colnames)
1497 if j !== nothing # Need to widen column type
1498 local newcols
1499 let i = i, j = j, outcols=outcols, rows=rows # Workaround for julia#15276
1500 newcols = ntuple(length(outcols)) do k
1501 S = eltype(rows isa AbstractDataFrame ? rows[!, k] : rows[k])
1502 T = eltype(outcols[k])
1503 U = promote_type(S, T)
1504 if S <: T || U <: T
1505 outcols[k]
1506 else
1507 copyto!(Tables.allocatecolumn(U, length(outcols[k])), outcols[k])
1508 end
1509 end
1510 end
1511 return _combine_tables_with_first!(rows, newcols, idx, i, j,
1512 f, gd, incols, colnames, firstmulticol)
1513 end
1514 append!(idx, Iterators.repeated(gdidx[starts[i]], _nrow(rows)))
1515 end
1516 return outcols, colnames
1517 end
