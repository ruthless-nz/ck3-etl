617 # use a dict because we dont know which measures the user wants
618 # Outside of the `describe` function due to something with 0.7
619 function get_stats(col::AbstractVector, stats::AbstractVector{Symbol})
620 d = Dict{Symbol, Any}()
621
622 if :q25 in stats || :median in stats || :q75 in stats
623 q = try quantile(col, [.25, .5, .75]) catch; (nothing, nothing, nothing) end
624 d[:q25] = q[1]
625 d[:median] = q[2]
626 d[:q75] = q[3]
627 end
628
629 if :min in stats || :max in stats
630 ex = try extrema(col) catch; (nothing, nothing) end
631 d[:min] = ex[1]
632 d[:max] = ex[2]
633 end
634
635 if :mean in stats || :std in stats
636 m = try mean(col) catch end
637 # we can add non-necessary things to d, because we choose what we need
638 # in the main function
639 d[:mean] = m
640
641 if :std in stats
642 d[:std] = try std(col, mean = m) catch end
643 end
644 end
645
646 if :nunique in stats
647 if eltype(col) <: Real
648 d[:nunique] = nothing
649 else
650 d[:nunique] = try length(unique(col)) catch end
651 end
652 end
653
654 return d
655 end

1155 @inbounds for g_row in gslots
1156 (g_row > 0) && (res[g_row] = false)
1157 end
1158 return res

1283 vcat(dfs::AbstractDataFrame...;
1284 cols::Union{Symbol, AbstractVector{Symbol},
1285 AbstractVector{<:AbstractString}}=:setequal)
1286
1287 Vertically concatenate `AbstractDataFrame`s.
1288
1289 The `cols` keyword argument determines the columns of the returned data frame:
1290
1291 * `:setequal`: require all data frames to have the same column names disregarding
1292 order. If they appear in different orders, the order of the first provided data
1293 frame is used.
1294 * `:orderequal`: require all data frames to have the same column names and in the
1295 same order.
1296 * `:intersect`: only the columns present in *all* provided data frames are kept.
1297 If the intersection is empty, an empty data frame is returned.
1298 * `:union`: columns present in *at least one* of the provided data frames are kept.
1299 Columns not present in some data frames are filled with `missing` where necessary.
1300 * A vector of `Symbol`s or strings: only listed columns are kept.
1301 Columns not present in some data frames are filled with `missing` where necessary.
1302
1303 The order of columns is determined by the order they appear in the included data
1304 frames, searching through the header of the first data frame, then the second,
1305 etc.
1306
1307 The element types of columns are determined using `promote_type`,
1308 as with `vcat` for `AbstractVector`s.

1310 `vcat` ignores empty data frames, making it possible to initialize an empty
1311 data frame at the beginning of a loop and `vcat` onto it.

1382 function _vcat(dfs::AbstractVector{<:AbstractDataFrame};
1383 cols::Union{Symbol, AbstractVector{Symbol},
1384 AbstractVector{<:AbstractString}}=:setequal)
1385
1386 isempty(dfs) && return DataFrame()
1387 # Array of all headers
1388 allheaders = map(names, dfs)
1389 # Array of unique headers across all data frames
1390 uniqueheaders = unique(allheaders)
1391 # All symbols present across all headers
1392 unionunique = union(uniqueheaders...)
1393 # List of symbols present in all dataframes
1394 intersectunique = intersect(uniqueheaders...)
1395
1396 if cols === :orderequal
1397 header = unionunique
1398 if length(uniqueheaders) > 1
1399 throw(ArgumentError("when `cols=:orderequal` all data frames need to " *
1400 "have the same column names and be in the same order"))
1401 end
1402 elseif cols === :setequal || cols === :equal
1403 if cols === :equal
1404 Base.depwarn("`cols=:equal` is deprecated." *
1405 "Use `:setequal` instead.", :vcat)
1406 end
1407
1408 header = unionunique
1409 coldiff = setdiff(unionunique, intersectunique)
1410
1411 if !isempty(coldiff)
1412 # if any DataFrames are a full superset of names, skip them
1413 filter!(u -> !issetequal(u, header), uniqueheaders)
1414 estrings = map(enumerate(uniqueheaders)) do (i, head)
1415 matching = findall(h -> head == h, allheaders)
1416 headerdiff = setdiff(coldiff, head)
1417 cols = join(headerdiff, ", ", " and ")
1418 args = join(matching, ", ", " and ")
1419 return "column(s) $cols are missing from argument(s) $args"
1420 end
1421 throw(ArgumentError(join(estrings, ", ", ", and ")))
1422 end
1423 elseif cols === :intersect
1424 header = intersectunique
1425 elseif cols === :union
1426 header = unionunique
1427 elseif cols isa Symbol
1428 throw(ArgumentError("Invalid `cols` value :$cols. " *
1429 "Only `:orderequal`, `:setequal`, `:intersect`, " *
1430 "`:union`, or a vector of column names is allowed."))
1431 elseif cols isa AbstractVector{Symbol}
1432 header = cols
1433 else
1434 @assert cols isa AbstractVector{<:AbstractString}
1435 header = Symbol.(cols)
1436 end
1437
1438 length(header) == 0 && return DataFrame()
1439 all_cols = Vector{AbstractVector}(undef, length(header))
1440 for (i, name) in enumerate(header)
1441 newcols = map(dfs) do df
1442 if hasproperty(df, name)
1443 return df[!, name]
1444 else
1445 Iterators.repeated(missing, nrow(df))
1446 end
1447 end
1448
1449 lens = map(length, newcols)
1450 T = mapreduce(eltype, promote_type, newcols)
1451 all_cols[i] = Tables.allocatecolumn(T, sum(lens))
1452 offset = 1
1453 for j in 1:length(newcols)
1454 copyto!(all_cols[i], offset, newcols[j])
1455 offset += lens[j]
1456 end
1457 end
1458 return DataFrame(all_cols, header, copycols=false)
1459 end

1858 function flatten(df::AbstractDataFrame,
1859 cols::Union{ColumnIndex, MultiColumnIndex})
1860 _check_consistency(df)
1861
1862 idxcols = index(df)[cols]
1863 isempty(idxcols) && return copy(df)
1864 col1 = first(idxcols)
1865 lengths = length.(df[!, col1])
1866 for col in idxcols
1867 v = df[!, col]
1868 if any(x -> length(x[1]) != x[2], zip(v, lengths))
1869 r = findfirst(x -> x != 0, length.(v) .- lengths)
1870 colnames = _names(df)
1871 throw(ArgumentError("Lengths of iterables stored in columns :$(colnames[col1])" *
1872 " and :$(colnames[col]) are not the same in row $r"))
1873 end
1874 end
1875
1876 new_df = similar(df[!, Not(cols)], sum(lengths))
1877 for name in _names(new_df)
1878 repeat_lengths!(new_df[!, name], df[!, name], lengths)
1879 end
1880 length(idxcols) > 1 && sort!(idxcols)
1881 for col in idxcols
1882 col_to_flatten = df[!, col]
1883 flattened_col = col_to_flatten isa AbstractVector{<:AbstractVector} ?
1884 reduce(vcat, col_to_flatten) :
1885 collect(Iterators.flatten(col_to_flatten))
1886
1887 insertcols!(new_df, col, _names(df)[col] => flattened_col)
1888 end
1889
1890 return new_df
1891 end
