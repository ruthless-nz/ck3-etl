266 # Fuzzy matching rules:
267 # 1. ignore case
268 # 2. maximum Levenshtein distance is 2
269 # 3. always show matches with 0 difference (wrong case)
270 # 4. on top of 3. do not show more than 8 matches in total
271 # Returns candidates ordered by (distance, name) pair
272 function fuzzymatch(l::Dict{Symbol, Int}, idx::Symbol)
273 idxs = uppercase(string(idx))
274 dist = [(REPL.levenshtein(uppercase(string(x)), idxs), x) for x in keys(l)]
275 sort!(dist)
276 c = [count(x -> x[1] <= i, dist) for i in 0:2]
277 maxd = max(0, searchsortedlast(c, 8) - 1)
278 return [s for (d, s) in dist if d <= maxd]
279 end
